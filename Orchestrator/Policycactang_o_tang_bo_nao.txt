class StructuredLogger:
    """JSON-line logger — mỗi dòng là 1 JSON object."""

    def __init__(self, name: str, level: int = logging.INFO):
        self._log = logging.getLogger(name)
        self._log.setLevel(level)
        if not self._log.handlers:
            h = logging.StreamHandler()
            h.setFormatter(logging.Formatter("%(message)s"))
            self._log.addHandler(h)

    def _emit(self, level: str, msg: str, **kw: Any) -> None:
        record = {
            "ts": datetime.now(timezone.utc).isoformat(),
            "level": level,
            "service": "orchestrator",
            "msg": msg,
            **kw,
        }
        method = getattr(self._log, level, self._log.info)
        method(json.dumps(record, ensure_ascii=False, default=str))

    def info(self, msg: str, **kw: Any) -> None:
        self._emit("info", msg, **kw)

    def warning(self, msg: str, **kw: Any) -> None:
        self._emit("warning", msg, **kw)

    def error(self, msg: str, **kw: Any) -> None:
        self._emit("error", msg, **kw)

    def critical(self, msg: str, **kw: Any) -> None:
        self._emit("critical", msg, **kw)


log = StructuredLogger("orchestrator")

def utc_now() -> datetime:
    return datetime.now(timezone.utc)

def brain_live_log(
    event: str,
    *,
    actor: str = "",
    host: str = "",
    role: str = "",
    target: str = "",
    action: str = "",
    status: str = "ok",
    dataset: str = "",
    detail: str = "",
    command_id: str = "",
) -> None:
    """
    Live log cho màn hình Tier3/Orchestrator.
    Dùng để show cho người xem thấy bộ não đang nhận yêu cầu,
    kiểm tra host, map role, cấp quyền, nhận event và ghi command.

    Tuyệt đối không ghi SECRET_ID thật, Vault token, DB password, encryption key.
    """
    try:
        ts = utc_now().isoformat()

        safe_detail = str(detail or "")

        sensitive_words = [
            "secret_id",
            "client_token",
            "vault_token",
            "password",
            "enc_key",
            "VAULT_TOKEN",
            "DB_PASS",
            "SECRET_ID",
            "VM_TOKEN",
            "ENC_KEY",
        ]

        for word in sensitive_words:
            safe_detail = safe_detail.replace(word, f"{word}_hidden")

        record = {
            "timestamp": ts,
            "service": "orchestrator-brain",
            "event": event,
            "actor": actor or "-",
            "host": host or "-",
            "role": role or "-",
            "target": target or "-",
            "action": action or "-",
            "status": status or "-",
            "dataset": dataset or "-",
            "detail": safe_detail,
            "command_id": command_id or "-",
        }

        tier_label = (actor or "orchestrator").upper()
        status_label = (status or "ok").upper()
        event_label = event.replace("_", " ").title()

        dataset_label = ""
        if dataset:
            dataset_label = f" | dataset={dataset}"

        detail_label = ""
        if safe_detail:
            detail_label = f" | {safe_detail}"

        line = (
            f"{ts} [ORCHESTRATOR-BRAIN] "
            f"{tier_label} | {status_label} | {event_label}"
            f"{dataset_label}"
            f"{detail_label}"
        )

        with BRAIN_LOG_LOCK:
            BRAIN_LIVE_LOG.parent.mkdir(parents=True, exist_ok=True)

            with BRAIN_LIVE_LOG.open("a", encoding="utf-8") as f:
                f.write(line + "\n")

            BRAIN_LATEST_FILE.write_text(
                json.dumps(record, indent=2, ensure_ascii=False),
                encoding="utf-8",
            )

        log.info(
            "brain_live_log",
            event=event,
            actor=actor,
            host=host,
            role=role,
            target=target,
            action=action,
            status=status,
            dataset=dataset,
            detail=safe_detail,
            command_id=command_id,
        )

    except Exception as exc:
        log.warning("brain_live_log_failed", error=str(exc))


def _get_db() -> sqlite3.Connection:
    if not getattr(_db_local, "conn", None):
        conn = sqlite3.connect(str(DB_PATH), check_same_thread=True, timeout=10)
        conn.row_factory = sqlite3.Row
        conn.execute("PRAGMA journal_mode=WAL")
        conn.execute("PRAGMA synchronous=NORMAL")
        conn.execute("PRAGMA foreign_keys=ON")
        conn.execute("PRAGMA temp_store=MEMORY")
        _db_local.conn = conn
    return _db_local.conn


def _ensure_column(conn: sqlite3.Connection, table: str, column: str, ddl: str) -> None:
    cols = {
        row["name"]
        for row in conn.execute(f"PRAGMA table_info({table})").fetchall()
    }
    if column not in cols:
        conn.execute(f"ALTER TABLE {table} ADD COLUMN {column} {ddl}")


def init_db() -> None:
    conn = _get_db()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS security_events (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            event_id TEXT NOT NULL UNIQUE,
            timestamp TEXT NOT NULL,
            host TEXT NOT NULL DEFAULT 'unknown',
            event_type TEXT NOT NULL,
            severity TEXT NOT NULL DEFAULT 'medium',
            dataset TEXT NOT NULL DEFAULT '',
            detail TEXT NOT NULL DEFAULT '',
            source TEXT NOT NULL DEFAULT 'tier2',
            dedup_key TEXT NOT NULL DEFAULT '',
            created_at TEXT NOT NULL
        );

        CREATE INDEX IF NOT EXISTS idx_events_type_ts
            ON security_events (event_type, timestamp);

        CREATE INDEX IF NOT EXISTS idx_events_ts
            ON security_events (timestamp);

        CREATE INDEX IF NOT EXISTS idx_events_dedup
            ON security_events (dedup_key);

        CREATE TABLE IF NOT EXISTS mode_history (
            id              INTEGER PRIMARY KEY AUTOINCREMENT,
            operation_mode  TEXT NOT NULL,
            changed_at      TEXT NOT NULL,
            changed_by      TEXT NOT NULL,
            reason          TEXT NOT NULL,
            source          TEXT NOT NULL DEFAULT '',
            trigger_event   TEXT NOT NULL DEFAULT '',
            rule_id         TEXT NOT NULL DEFAULT '',
            auto            INTEGER NOT NULL DEFAULT 1
        );
    """)

    _ensure_column(conn, "mode_history", "target_tier", "TEXT NOT NULL DEFAULT ''")
    _ensure_column(conn, "mode_history", "action", "TEXT NOT NULL DEFAULT ''")
    _ensure_column(conn, "mode_history", "cleanup_after_push", "INTEGER NOT NULL DEFAULT 0")
    _ensure_column(conn, "mode_history", "command_id", "TEXT NOT NULL DEFAULT ''")

    conn.commit()
    log.info("db_init_ok", db_path=str(DB_PATH))


def trim_old_events() -> int:
    cutoff = (utc_now() - timedelta(hours=EVENT_RETAIN_HOURS)).isoformat()
    conn = _get_db()
    cur = conn.execute(
        "DELETE FROM security_events WHERE timestamp < ?",
        (cutoff,),
    )
    conn.commit()
    deleted = cur.rowcount
    if deleted:
        log.info("events_trimmed", deleted=deleted, cutoff=cutoff)
    return deleted


def _dedup_key(host: str, event_type: str, dataset: str) -> str:
    bucket = int(time.time() // DEDUP_WINDOW_SEC)
    raw = f"{host}|{event_type}|{dataset}|{bucket}"
    return hashlib.sha256(raw.encode()).hexdigest()[:16]


def append_event(event: dict) -> tuple[bool, str]:
    dedup = _dedup_key(event["host"], event["event_type"], event["dataset"])
    conn = _get_db()

    row = conn.execute(
        "SELECT event_id FROM security_events WHERE dedup_key = ? LIMIT 1",
        (dedup,),
    ).fetchone()

    if row:
        log.info(
            "event_dedup_skipped",
            dedup_key=dedup,
            event_type=event["event_type"],
            host=event["host"],
        )
        return False, row["event_id"]

    event_id = str(uuid.uuid4())
    now_iso = utc_now().isoformat()

    conn.execute(
        """INSERT INTO security_events
           (event_id, timestamp, host, event_type, severity, dataset, detail, source, dedup_key, created_at)
           VALUES (?,?,?,?,?,?,?,?,?,?)""",
        (
            event_id,
            event["timestamp"],
            event["host"],
            event["event_type"],
            event["severity"],
            event.get("dataset", ""),
            event.get("detail", ""),
            event.get("source", "tier2"),
            dedup,
            now_iso,
        ),
    )
    conn.commit()
    return True, event_id


def query_events_in_window(event_type: str, cutoff_iso: str) -> list[sqlite3.Row]:
    conn = _get_db()
    return conn.execute(
        """SELECT * FROM security_events
           WHERE event_type = ? AND timestamp >= ?
           ORDER BY timestamp ASC""",
        (event_type, cutoff_iso),
    ).fetchall()

_vault_cache: dict[str, dict] = {}
_vault_cache_lock = threading.Lock()


def _get_vault_headers() -> dict:
    return {
        "X-Vault-Token": VAULT_TOKEN,
        "Content-Type": "application/json",
    }


def _vault_url(path: str) -> str:
    return f"{VAULT_ADDR.rstrip('/')}/v1/{path.lstrip('/')}"


def _vault_verify():
    # Nếu Vault đang dùng CA nội bộ, bắt buộc verify bằng CA này.
    # Nếu VAULT_CACERT rỗng thì dùng trust store mặc định của hệ điều hành.
    return VAULT_CACERT if VAULT_CACERT else True


@_cb
def _vault_get_path_raw(path: str) -> dict:
    r = requests.get(
        _vault_url(path),
        headers=_get_vault_headers(),
        timeout=HTTP_TIMEOUT,
        verify=_vault_verify(),
    )
    r.raise_for_status()
    body = r.json()
    return body.get("data", {}).get("data", {})


@_cb
def _vault_set_path_raw(path: str, data: dict) -> None:
    r = requests.post(
        _vault_url(path),
        headers=_get_vault_headers(),
        json={"data": data},
        timeout=HTTP_TIMEOUT,
        verify=_vault_verify(),
    )
    r.raise_for_status()


def vault_read_path(path: str, bypass_cache: bool = False) -> dict:
    with _vault_cache_lock:
        now = time.monotonic()
        item = _vault_cache.get(path)

        if not bypass_cache and item:
            if now < item["expires"]:
                return item["value"]

    try:
        val = _vault_get_path_raw(path)

        with _vault_cache_lock:
            _vault_cache[path] = {
                "value": val,
                "expires": time.monotonic() + VAULT_CACHE_TTL,
            }

        return val

    except CircuitBreakerError:
        log.error("vault_circuit_open", path=path)
        with _vault_cache_lock:
            item = _vault_cache.get(path)
            if item:
                return item["value"]
        raise

    except requests.RequestException as exc:
        log.error("vault_get_failed", path=path, error=str(exc))
        with _vault_cache_lock:
            item = _vault_cache.get(path)
            if item:
                log.warning("vault_serving_stale_cache", path=path)
                return item["value"]
        raise


def vault_write_path(path: str, data: dict) -> None:
    _vault_set_path_raw(path, data)

    with _vault_cache_lock:
        _vault_cache[path] = {
            "value": data,
            "expires": time.monotonic() + VAULT_CACHE_TTL,
        }

def set_tier5_control_for_command(
    *,
    mode: str,
    active_tier: str,
    reason: str,
    command_id: str,
    source: str,
    retention_days: int | None = None,
    retention_policy: str = "",
) -> dict:
    if mode == "normal":
        data = {
            "operation_mode": "normal",
            "accept_from_tier4": True,
            "accept_from_tier2": False,
            "immutable_lock": True,
            "retention_days": 30,
            "retention_policy": "normal_high_critical_from_tier4",
            "changed_by": "orchestrator-automation",
            "changed_at": utc_now().isoformat(),
            "reason": reason,
            "active_emergency_tier": "",
            "command_id": command_id,
            "source": source,
        }
    else:
        # Quan trọng:
        # - Tier4 failure → active_tier=tier2 và cleanup_after_push=false ở issue_exclusive_command()
        #   => Tier2 đẩy du_phong full lên Tier5 emergency, giữ 7 ngày.
        # - Security/ransomware emergency khác vẫn giữ dài hơn.
        if retention_days is None:
            retention_days = 90

        if not retention_policy:
            retention_policy = (
                "tier4_failure_full_emergency_7_days"
                if active_tier == "tier2" and retention_days == 7
                else "security_or_general_emergency_retention"
            )

        data = {
            "operation_mode": "emergency",
            "accept_from_tier4": active_tier == "tier4",
            "accept_from_tier2": active_tier == "tier2",
            "immutable_lock": True,
            "retention_days": retention_days,
            "retention_policy": retention_policy,
            "changed_by": "orchestrator-automation",
            "changed_at": utc_now().isoformat(),
            "reason": reason,
            "active_emergency_tier": active_tier,
            "command_id": command_id,
            "source": source,
        }

    vault_write_path(TIER5_CONTROL_PATH, data)
    log.info(
        "tier5_control_changed",
        mode=data["operation_mode"],
        accept_from_tier4=data["accept_from_tier4"],
        accept_from_tier2=data["accept_from_tier2"],
        active_emergency_tier=data["active_emergency_tier"],
        retention_days=data["retention_days"],
        retention_policy=data.get("retention_policy", ""),
        command_id=command_id,
    )
    brain_live_log(
        "TIER5_ACCEPTANCE_POLICY_UPDATED",
        actor="orchestrator",
        host="orchestrator",
        target="tier5",
        action="update_tier5_acceptance_policy",
        status=data["operation_mode"],
        command_id=command_id,
        detail=(
            f"accept_from_tier4={data['accept_from_tier4']}; "
            f"accept_from_tier2={data['accept_from_tier2']}; "
            f"immutable_lock={data['immutable_lock']}; "
            f"retention_days={data['retention_days']}; "
            f"policy={data.get('retention_policy', '')}"
        ),
    )
    return data


def set_control_in_vault(
    *,
    target_tier: str,
    mode: str,
    action: str,
    reason: str,
    source: str,
    trigger_event: str,
    rule_id: str = "",
    cleanup_after_push: bool = False,
    auto: bool = True,
    command_id: str | None = None,
) -> dict:
    if target_tier not in {"tier2", "tier4"}:
        raise ValueError("target_tier must be tier2 or tier4")

    if mode not in {"normal", "emergency"}:
        raise ValueError("mode must be normal or emergency")

    if target_tier == "tier2":
        # Tier2 emergency action có 2 kiểu hợp lệ:
        # 1) "push_du_phong_to_tier5"  -> emergency theo trạng thái hệ thống,
        #    Tier2 tự dùng DATASET_ID hiện tại đang verify/snapshot để push du_phong.
        # 2) "/data/tier5/incoming/tier2/emergency/<dataset_id>/" -> action cũ theo dataset cụ thể.
        if action not in {"none", "push_du_phong_to_tier5"} and not (
            action.startswith(TIER2_DU_PHONG_ACTION_PREFIX)
            and action.endswith("/")
        ):
            raise ValueError(f"invalid action '{action}' for {target_tier}")
    else:
        allowed_actions = {"none", "push_high_to_tier5", "push_full_to_tier5"}

        if action not in allowed_actions:
            raise ValueError(f"invalid action '{action}' for {target_tier}")

    changed_by = "orchestrator-automation" if auto else "orchestrator-admin"
    command_id = command_id or f"cmd-{utc_now().strftime('%Y%m%d%H%M%S')}-{uuid.uuid4().hex[:8]}"

    data = {
        "operation_mode": mode,
        "target_tier": target_tier,
        "action": action,
        "reason": reason,
        "cleanup_after_push": bool(cleanup_after_push),
        "command_id": command_id,
        "changed_at": utc_now().isoformat(),
        "changed_by": changed_by,
        "source": source,
        "trigger_event": trigger_event,
        "rule_id": rule_id,
        "auto": bool(auto),
    }

    vault_write_path(CONTROL_PATHS[target_tier], data)

    MODE_STATE_FILE.write_text(
        json.dumps(data, indent=2, ensure_ascii=False),
        encoding="utf-8",
    )

    append_mode_history(
        mode=mode,
        target_tier=target_tier,
        action=action,
        cleanup_after_push=cleanup_after_push,
        command_id=command_id,
        changed_by=changed_by,
        reason=reason,
        source=source,
        trigger_event=trigger_event,
        rule_id=rule_id,
        auto=auto,
    )

    log.info(
        "control_changed",
        target_tier=target_tier,
        mode=mode,
        action=action,
        cleanup_after_push=cleanup_after_push,
        reason=reason,
        source=source,
        trigger_event=trigger_event,
        rule_id=rule_id,
        command_id=command_id,
    )
    brain_live_log(
        "CONTROL_COMMAND_WRITTEN_TO_VAULT",
        actor="orchestrator",
        host="orchestrator",
        target=target_tier,
        action=action,
        status=mode,
        command_id=command_id,
        detail=(
            f"path={CONTROL_PATHS[target_tier]}; "
            f"reason={reason}; "
            f"cleanup_after_push={cleanup_after_push}; "
            f"trigger_event={trigger_event}; "
            f"rule_id={rule_id}"
        ),
    )
    return data


def suppress_opposite_tier(
    *,
    active_tier: str,
    reason: str,
    source: str,
    trigger_event: str,
    rule_id: str,
    auto: bool,
) -> dict:
    opposite = OPPOSITE_TIER[active_tier]

    if opposite == "tier4":
        action = "push_high_to_tier5"
        mode = "normal"
    else:
        action = "none"
        mode = "normal"

    return set_control_in_vault(
        target_tier=opposite,
        mode=mode,
        action=action,
        reason=f"suppressed to avoid double-push: {reason}",
        source=source,
        trigger_event=trigger_event,
        rule_id=rule_id,
        cleanup_after_push=False,
        auto=auto,
    )


def issue_exclusive_command(
    *,
    target_tier: str,
    mode: str,
    action: str,
    reason: str,
    source: str,
    trigger_event: str,
    rule_id: str,
    cleanup_after_push: bool,
    auto: bool = True,
) -> dict:
    state = set_control_in_vault(
        target_tier=target_tier,
        mode=mode,
        action=action,
        reason=reason,
        source=source,
        trigger_event=trigger_event,
        rule_id=rule_id,
        cleanup_after_push=cleanup_after_push,
        auto=auto,
    )

    suppress_opposite_tier(
        active_tier=target_tier,
        reason=reason,
        source=source,
        trigger_event=trigger_event,
        rule_id=rule_id,
        auto=auto,
    )

    # Retention policy:
    # - Tier4 failure: Orchestrator ra lệnh cho Tier2 dùng du_phong cứu hộ.
    #   Trường hợp này target_tier=tier2 và cleanup_after_push=false.
    #   Tier5 giữ full emergency 7 ngày.
    # - Security/ransomware/self anomaly: cleanup_after_push=true hoặc target_tier=tier4.
    #   Giữ emergency lâu hơn để điều tra.
    if mode == "emergency" and target_tier == "tier2" and cleanup_after_push is False:
        tier5_retention_days = 7
        tier5_retention_policy = "tier4_failure_full_emergency_7_days"
    elif mode == "emergency":
        tier5_retention_days = 90
        tier5_retention_policy = "security_or_general_emergency_retention"
    else:
        tier5_retention_days = None
        tier5_retention_policy = ""

    tier5_state = set_tier5_control_for_command(
        mode=mode,
        active_tier=target_tier,
        reason=reason,
        command_id=state["command_id"],
        source=source,
        retention_days=tier5_retention_days,
        retention_policy=tier5_retention_policy,
    )

    return {
        "active_tier": state,
        "tier5": tier5_state,
    }


def set_both_normal(reason: str, source: str, auto: bool) -> dict:
    tier4 = set_control_in_vault(
        target_tier="tier4",
        mode="normal",
        action="push_high_to_tier5",
        reason=reason,
        source=source,
        trigger_event="normal_reset",
        rule_id="NORMAL",
        cleanup_after_push=False,
        auto=auto,
    )

    tier2 = set_control_in_vault(
        target_tier="tier2",
        mode="normal",
        action="none",
        reason=reason,
        source=source,
        trigger_event="normal_reset",
        rule_id="NORMAL",
        cleanup_after_push=False,
        auto=auto,
    )

    tier5 = set_tier5_control_for_command(
        mode="normal",
        active_tier="tier4",
        reason=reason,
        command_id="normal-reset-tier5",
        source=source,
    )

    return {"tier4": tier4, "tier2": tier2, "tier5": tier5}

def _normalise_source(event: dict) -> str:
    src = (event.get("source") or "").strip().lower()
    host = (event.get("host") or "").strip().lower()

    if src in {"tier2", "tier4"}:
        return src

    if "tier2" in host:
        return "tier2"

    if "tier4" in host or "backupstore" in host:
        return "tier4"

    return src or "unknown"


def _event_detail(event: dict) -> str:
    return (event.get("detail") or "").strip().lower()


def decide_command_from_event(event: dict, rule_id: str = "DIRECT") -> dict | None:
    event_type = (event.get("event_type") or "").strip()
    severity = (event.get("severity") or "medium").strip().lower()
    source = _normalise_source(event)
    detail = _event_detail(event)

    # ── Tier4 sụp hoặc Tier2 nhìn thấy Tier4 không reachable → Tier2 cứu hộ, không xóa du_phong
    if event_type in TIER4_FATAL_EVENTS_TO_TIER2:
        return {
            "target_tier": "tier2",
            "mode": "emergency",
            "action": "push_du_phong_to_tier5",
            "reason": f"Tier4 unavailable, Tier2 must use du_phong. event={event_type}",
            "cleanup_after_push": False,
            "rule_id": rule_id,
        }

    if source == "tier2" and event_type in {"agent_unreachable", "backup_failure"}:
        if "tier4" in detail or "backupstore" in detail or "rsync_to_backupstore" in detail:
            return {
                "target_tier": "tier2",
                "mode": "emergency",
                "action": "push_du_phong_to_tier5",
                "reason": f"Tier2 observed Tier4/BackupStore failure. detail={detail}",
                "cleanup_after_push": False,
                "rule_id": rule_id,
            }

    # ── Tier2 sụp hoặc Tier4 nhìn thấy Tier2 không reachable → Tier4 cứu hộ, không xóa dữ liệu Tier4
    if event_type in TIER2_FATAL_EVENTS_TO_TIER4:
        return {
            "target_tier": "tier4",
            "mode": "emergency",
            "action": "push_full_to_tier5",
            "reason": f"Tier2 unavailable, Tier4 must promote full dataset. event={event_type}",
            "cleanup_after_push": False,
            "rule_id": rule_id,
        }

    if source == "tier4" and event_type in {"agent_unreachable", "backup_failure"}:
        if "tier2" in detail or "tier2-snapshot" in detail:
            return {
                "target_tier": "tier4",
                "mode": "emergency",
                "action": "push_full_to_tier5",
                "reason": f"Tier4 observed Tier2 failure. detail={detail}",
                "cleanup_after_push": False,
                "rule_id": rule_id,
            }

    # ── Tier4 fatal nội bộ: restore/extract/repo hỏng → Tier4 có thể không push được, Tier2 cứu hộ.
    if source == "tier4" and event_type in {"restore_failure", "tier4_repo_corrupt"}:
        return {
            "target_tier": "tier2",
            "mode": "emergency",
            "action": "push_du_phong_to_tier5",
            "reason": f"Tier4 cannot restore/process safely. event={event_type} detail={detail}",
            "cleanup_after_push": False,
            "rule_id": rule_id,
        }

    if source == "tier4" and event_type == "backup_failure":
        if any(x in detail for x in [
            "cannot_connect_kopia",
            "no_snapshot_found",
            "kopia_restore_failed",
            "tar_extract_failed",
            "rsync_promote_bundle_to_tier5_failed",
            "cannot_ssh_to_tier5",
        ]):
            return {
                "target_tier": "tier2",
                "mode": "emergency",
                "action": "push_du_phong_to_tier5",
                "reason": f"Tier4 backup/promote path failed. detail={detail}",
                "cleanup_after_push": False,
                "rule_id": rule_id,
            }

    # ── Tier4 bị nghi ngờ tấn công nhưng còn chạy: Tier4 tự push full và xóa local sau push.
    if source == "tier4" and (
        event_type in SELF_ANOMALY_EVENTS
        or severity == "critical"
    ):
        return {
            "target_tier": "tier4",
            "mode": "emergency",
            "action": "push_full_to_tier5",
            "reason": f"Tier4 anomaly/attack suspected. event={event_type} detail={detail}",
            "cleanup_after_push": True,
            "rule_id": rule_id,
        }

    # ── Tier2 bị nghi ngờ tấn công nhưng còn chạy: Tier2 push du_phong và xóa du_phong sau push.
    if source == "tier2" and (
        event_type in SELF_ANOMALY_EVENTS
        or severity == "critical"
    ):
        return {
            "target_tier": "tier2",
            "mode": "emergency",
            "action": "push_du_phong_to_tier5",
            "reason": f"Tier2 anomaly/attack suspected. event={event_type} detail={detail}",
            "cleanup_after_push": True,
            "rule_id": rule_id,
        }

    # ── Tier2 backup failure nội bộ không liên quan Tier4: coi là Tier2 bất thường.
    if source == "tier2" and event_type == "backup_failure":
        return {
            "target_tier": "tier2",
            "mode": "emergency",
            "action": "push_du_phong_to_tier5",
            "reason": f"Tier2 backup path abnormal. detail={detail}",
            "cleanup_after_push": True,
            "rule_id": rule_id,
        }

    return None

_eval_lock = threading.Lock()


def _check_single_rule(rule: dict, now: datetime) -> list[sqlite3.Row]:
    cutoff = (now - timedelta(minutes=rule["window_min"])).isoformat()
    rows = query_events_in_window(rule["event_type"], cutoff)
    return [
        r for r in rows
        if severity_ok(r["severity"], rule["severity_at_least"])
    ]


def _check_composite_rule(rule: dict, now: datetime) -> list[sqlite3.Row]:
    cutoff = (now - timedelta(minutes=rule["window_min"])).isoformat()
    matched: list[sqlite3.Row] = []

    for et in rule["composite"]:
        rows = query_events_in_window(et, cutoff)
        qualified = [
            r for r in rows
            if severity_ok(r["severity"], rule["severity_at_least"])
        ]
        if not qualified:
            return []
        matched.extend(qualified)

    return matched


def evaluate_rules(latest_event: dict | None = None) -> dict:
    with _eval_lock:
        now = utc_now()
        trim_old_events()

        # Ưu tiên quyết định trực tiếp từ event mới nhất.
        if latest_event:
            direct_cmd = decide_command_from_event(latest_event, rule_id="DIRECT")
            if direct_cmd:
                try:
                    new_state = issue_exclusive_command(
                        target_tier=direct_cmd["target_tier"],
                        mode=direct_cmd["mode"],
                        action=direct_cmd["action"],
                        reason=direct_cmd["reason"],
                        source="orchestrator-auto",
                        trigger_event=latest_event["event_type"],
                        rule_id=direct_cmd["rule_id"],
                        cleanup_after_push=direct_cmd["cleanup_after_push"],
                        auto=True,
                    )
                except Exception as exc:
                    log.critical(
                        "control_set_failed",
                        error=str(exc),
                        tb=traceback.format_exc(),
                    )
                    return {"triggered": False, "error": str(exc)}

                if HAS_PROM:
                    _prom_emergency_total.labels(
                        rule_id=direct_cmd["rule_id"],
                        target_tier=direct_cmd["target_tier"],
                    ).inc()

                log.critical(
                    "CONTROL_TRIGGERED_DIRECT",
                    target_tier=direct_cmd["target_tier"],
                    action=direct_cmd["action"],
                    reason=direct_cmd["reason"],
                    cleanup_after_push=direct_cmd["cleanup_after_push"],
                )

                return {
                    "triggered": True,
                    "decision_type": "direct_policy",
                    "target_tier": direct_cmd["target_tier"],
                    "action": direct_cmd["action"],
                    "cleanup_after_push": direct_cmd["cleanup_after_push"],
                    "new_state": new_state,
                }

        # Sau đó mới xét rule threshold.
        for rule in RULES:
            rule_id = rule["rule_id"]

            if "composite" in rule:
                matched = _check_composite_rule(rule, now)
            else:
                matched = _check_single_rule(rule, now)

            if not matched:
                continue

            if "composite" not in rule:
                if len(matched) < rule["threshold"]:
                    continue

            latest_row = matched[-1]
            synthetic_event = {
                "timestamp": latest_row["timestamp"],
                "host": latest_row["host"],
                "event_type": latest_row["event_type"],
                "severity": latest_row["severity"],
                "dataset": latest_row["dataset"],
                "detail": latest_row["detail"],
                "source": latest_row["source"],
            }

            cmd = decide_command_from_event(synthetic_event, rule_id=rule_id)
            if not cmd:
                continue

            try:
                new_state = issue_exclusive_command(
                    target_tier=cmd["target_tier"],
                    mode=cmd["mode"],
                    action=cmd["action"],
                    reason=f'{rule["reason"]}. {cmd["reason"]}',
                    source="orchestrator-auto",
                    trigger_event=synthetic_event["event_type"],
                    rule_id=rule_id,
                    cleanup_after_push=cmd["cleanup_after_push"],
                    auto=True,
                )
            except Exception as exc:
                log.critical(
                    "control_set_failed",
                    rule_id=rule_id,
                    error=str(exc),
                    tb=traceback.format_exc(),
                )
                return {"triggered": False, "error": str(exc)}

            if HAS_PROM:
                _prom_emergency_total.labels(
                    rule_id=rule_id,
                    target_tier=cmd["target_tier"],
                ).inc()

            log.critical(
                "CONTROL_TRIGGERED_RULE",
                rule_id=rule_id,
                target_tier=cmd["target_tier"],
                action=cmd["action"],
                matched_count=len(matched),
                cleanup_after_push=cmd["cleanup_after_push"],
            )

            return {
                "triggered": True,
                "decision_type": "threshold_rule",
                "rule_id": rule_id,
                "matched_count": len(matched),
                "target_tier": cmd["target_tier"],
                "action": cmd["action"],
                "cleanup_after_push": cmd["cleanup_after_push"],
                "new_state": new_state,
            }

        return {"triggered": False}

_rate_store: dict[str, list[float]] = defaultdict(list)
_rate_lock = threading.Lock()


def _is_rate_limited(ip: str) -> bool:
    now = time.monotonic()
    cutoff = now - 60.0

    with _rate_lock:
        hits = _rate_store[ip]
        _rate_store[ip] = [t for t in hits if t > cutoff]

        if len(_rate_store[ip]) >= RATE_LIMIT_RPM:
            return True

        _rate_store[ip].append(now)
        return False

def _get_client_ip() -> str:
    xff = request.headers.get("X-Forwarded-For", "")
    return xff.split(",")[0].strip() if xff else (request.remote_addr or "unknown")


def require_auth(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        ip = _get_client_ip()

        if _is_rate_limited(ip):
            log.warning("rate_limited", ip=ip, endpoint=request.path)
            return jsonify({"error": "rate limit exceeded", "retry_after_sec": 60}), 429

        if ALLOWED_IPS and ip not in ALLOWED_IPS:
            log.warning("ip_blocked", ip=ip, endpoint=request.path)
            return jsonify({"error": "forbidden"}), 403

        if ORCHESTRATOR_API_KEY:
            provided = request.headers.get("X-API-Key", "").strip()
            if not provided or provided != ORCHESTRATOR_API_KEY:
                log.warning("auth_failed", ip=ip, endpoint=request.path)
                return jsonify({"error": "unauthorized"}), 401

        return f(*args, **kwargs)

    return decorated


@app.before_request
def _before():
    g.start_time = time.monotonic()
    g.request_id = str(uuid.uuid4())


@app.after_request
def _after(response):
    elapsed = time.monotonic() - getattr(g, "start_time", time.monotonic())
    response.headers["X-Request-ID"] = getattr(g, "request_id", "")
    response.headers["X-Response-Time-Ms"] = f"{elapsed * 1000:.1f}"

    if HAS_PROM:
        _prom_request_total.labels(
            method=request.method,
            endpoint=request.path,
            status=response.status_code,
        ).inc()
        _prom_request_latency.labels(endpoint=request.path).observe(elapsed)
        if HAS_BREAKER:
            _prom_vault_circuit_state.set(1 if vault_breaker.current_state == "open" else 0)

    return response


def ok(data: Any, code: int = 200):
    return jsonify(data), code

def run_local_cmd(cmd: list[str], timeout: int = 30) -> dict:
    try:
        p = subprocess.run(
            cmd,
            text=True,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            timeout=timeout,
        )
        return {
            "ok": p.returncode == 0,
            "returncode": p.returncode,
            "stdout": p.stdout.strip(),
            "stderr": p.stderr.strip(),
        }
    except Exception as exc:
        return {
            "ok": False,
            "returncode": -1,
            "stdout": "",
            "stderr": str(exc),
        }

        "event_id": event_id,

        "automation": result,
    }, 202)

def read_tier_status(tier: str) -> dict:
    if tier not in TIER_STATUS_FILES:
        return {
            "tier": tier,
            "status": "invalid_tier",
            "message": "Unknown tier",
        }

    status_file = TIER_STATUS_FILES[tier]
    r = ssh_cmd(tier, f"cat {status_file}", timeout=8)

    if not r["ok"]:
        msg = r["stderr"] or r["stdout"] or "No status file"

        # Đổi lỗi SSH thô thành câu dễ hiểu cho dashboard.
        if "No route to host" in msg or "connect to host" in msg or "Connection timed out" in msg:
            return {
                "tier": tier,
                "status": "offline",
                "message": f"{tier.upper()} VM chưa bật hoặc Orchestrator chưa SSH được tới tầng này.",
                "raw_error": msg,
            }

        return {
            "tier": tier,
            "status": "unknown",
            "message": msg,
        }

    try:
        data = json.loads(r["stdout"])
        if "tier" not in data:
            data["tier"] = tier
        return data
    except Exception:
        return {
            "tier": tier,
            "status": "invalid_json",
            "raw": r["stdout"],
        }

def run_tier_service(tier: str, timeout: int = 1800, dataset_id: str = "") -> dict:
    if tier not in TIER_NODES:
        return {
            "ok": False,
            "tier": tier,
            "stderr": "unknown tier",
            "stdout": "",
            "returncode": -1,
        }
    # Safety gate:
    # Nếu Orchestrator đang ở emergency route Tier2 -> Tier5
    # thì Tier4 được xem là không đáng tin / đang bị cô lập.
    # Không cho chạy Tier4 cho tới khi admin reset-normal.
    if tier == "tier4" and _tier5_accepts_tier2_emergency():
        return {
            "ok": False,
            "tier": tier,
            "service": TIER_NODES[tier]["service"],
            "stdout": "",
            "stderr": "Tier4 is blocked: emergency route Tier2 -> Tier5 is active. Reset normal before running Tier4.",
            "returncode": 99,
        }


    service = TIER_NODES[tier]["service"]

    if dataset_id:
        safe_dataset = shlex.quote(dataset_id)
        cmd = (
            f"sudo -n /usr/bin/systemctl set-environment PIPELINE_DATASET_ID={safe_dataset} && "
            f"sudo -n /usr/bin/systemctl start {service}; "
            f"rc=$?; "
            f"sudo -n /usr/bin/systemctl unset-environment PIPELINE_DATASET_ID; "
            f"exit $rc"
        )
    else:
        cmd = f"sudo -n /usr/bin/systemctl start {service}"

    r = ssh_cmd(tier, cmd, timeout=timeout)

    return {
        "ok": r["ok"],
        "tier": tier,
        "service": service,
        "stdout": r["stdout"],
        "stderr": r["stderr"],
        "returncode": r["returncode"],
    }

@app.route("/pipeline/normal/start", methods=["POST"])
@require_auth
def start_normal_pipeline():
    body = request.get_json(silent=True) or {}
    reason = body.get("reason", "normal pipeline started from dashboard")

    pipeline_result = {
        "status": "running",
        "mode": "normal",
        "started_at": utc_now().isoformat(),
        "reason": reason,
        "steps": [],
    }

    # 1. Reset normal mode trước khi chạy.
    try:
        normal_state = set_both_normal(
            reason=reason,
            source=_get_client_ip(),
            auto=False,
        )
        pipeline_result["normal_state"] = normal_state
    except Exception as exc:
        return ok({
            "status": "failed",
            "failed_at": "reset-normal",
            "error": str(exc),
            "pipeline": pipeline_result,
        }, 500)

    # 2. Chạy từng tầng theo thứ tự.
    # Tier1 tạo dataset mới trước.
    # Sau khi Tier1 chạy xong, Orchestrator đọc dataset_id của Tier1
    # rồi truyền dataset_id đó xuống Tier2, Tier4, Tier5.
    active_dataset_id = ""

    for tier in ["tier1", "tier2", "tier4", "tier5"]:
        step = {
            "tier": tier,
            "started_at": utc_now().isoformat(),
        }

        result = run_tier_service(
            tier,
            dataset_id=active_dataset_id,
        )

        step.update(result)
        step["finished_at"] = utc_now().isoformat()

        pipeline_result["steps"].append(step)

        if not result["ok"]:
            pipeline_result["status"] = "failed"
            pipeline_result["failed_at"] = tier
            pipeline_result["finished_at"] = utc_now().isoformat()
            return ok(pipeline_result, 500)

        if tier == "tier1":
            tier1_status = read_tier_status("tier1")
            active_dataset_id = tier1_status.get("dataset_id", "")

            if not active_dataset_id:
                pipeline_result["status"] = "failed"
                pipeline_result["failed_at"] = "tier1"
                pipeline_result["finished_at"] = utc_now().isoformat()
                pipeline_result["error"] = "Tier1 completed but did not produce dataset_id"
                return ok(pipeline_result, 500)

            pipeline_result["dataset_id"] = active_dataset_id

    pipeline_result["status"] = "success"
    pipeline_result["finished_at"] = utc_now().isoformat()

    return ok(pipeline_result)

def _restore_validate_id(label: str, value: str) -> str:
    value = (value or "").strip()

    if not re.match(r"^[a-zA-Z0-9_-]{1,128}$", value):
        raise ValueError(f"{label} invalid: {value}")

    return value


def _restore_new_id(prefix: str = "restore") -> str:
    return f"{prefix}-{utc_now().strftime('%Y%m%d-%H%M%S')}-{uuid.uuid4().hex[:6]}"


def _restore_expires_at(minutes: int = 60) -> str:
    return (utc_now() + timedelta(minutes=minutes)).isoformat()


def _restore_bool(v: Any, default: bool = False) -> bool:
    if v is None:
        return default

    if isinstance(v, bool):
        return v

    return str(v).strip().lower() in {"true", "1", "yes", "y", "on"}


def _restore_int(v: Any, default: int) -> int:
    try:
        return int(v)
    except Exception:
        return default


def _restore_get_target(target_id: str, source_tier: str, body: dict | None = None) -> dict:
    body = body or {}
    source_tier = (source_tier or "").strip().lower()
    target_id = (target_id or "tier1-restore-agent").strip()

    if target_id not in RESTORE_TARGETS:
        raise ValueError(f"unknown restore target: {target_id}")

    target = dict(RESTORE_TARGETS[target_id])

    # ============================================================
    # Tier5 restore HIGH:
    #   - Bắt buộc dùng drop-only account restore_drop
    #   - Bắt buộc drop-zone /restore/incoming/tier5
    # ============================================================
    if source_tier == "tier5":

                f"Tier5 restore must use restricted drop user {tier5_drop_user}, "
                f"got {body['target_agent_user']}"

            raise ValueError(
                f"Tier5 restore must use drop-zone {tier5_drop_dir}, "
                f"got {body['target_incoming_dir']}"
            )

        target["target_agent_user"] = tier5_drop_user
        target["target_incoming_dir"] = tier5_drop_dir

        return target

    # ============================================================
    # Tier4 restore:
    #   - Bắt buộc dùng drop-only account restore_tier4_drop
    #   - Bắt buộc drop-zone /restore/incoming/tier4
    # ============================================================

        tier4_drop_dir = _optional_env("RESTORE_TIER1_TIER4_INCOMING_DIR", "/restore/incoming/tier4")

            "target_agent_host",

            if body.get(key):
                target[key] = str(body[key]).strip()

        if body.get("target_agent_user") and str(body["target_agent_user"]).strip() != tier4_drop_user:
            raise ValueError(
                f"Tier4 restore must use restricted drop user {tier4_drop_user}, "
                f"got {body['target_agent_user']}"
            )

        if body.get("target_incoming_dir") and str(body["target_incoming_dir"]).strip() != tier4_drop_dir:
            raise ValueError(
                f"Tier4 restore must use drop-zone {tier4_drop_dir}, "
                f"got {body['target_incoming_dir']}"
            )

        target["target_agent_user"] = tier4_drop_user
        target["target_incoming_dir"] = tier4_drop_dir

        return target

    # Fallback nếu sau này có source_tier khác.
    for key in [
        "target_agent_host",
        "target_agent_ip",
        "target_agent_user",
        "target_agent_mac",
        "target_agent_ssh_fingerprint",
    ]:
        if body.get(key):
            target[key] = str(body[key]).strip()

    target_incoming_dir = body.get("target_incoming_dir")
    if target_incoming_dir:
        target["target_incoming_dir"] = str(target_incoming_dir).strip()
    else:
        target["target_incoming_dir"] = f"{target['target_incoming_base'].rstrip('/')}/{source_tier}"

    return target

def _restore_read_release_status(source_tier: str) -> dict:
    status_file = RESTORE_STATUS_FILES.get(source_tier, "")

    if not status_file:
        return {
            "source_tier": source_tier,
            "status": "unknown",
            "message": "restore status file is not configured",
        }

    r = ssh_cmd(
        source_tier,
        f"cat {shlex.quote(status_file)}",
        timeout=10,
    )

    if not r["ok"]:
        return {
            "source_tier": source_tier,
            "status": "unknown",
            "message": r["stderr"] or r["stdout"] or f"Cannot read {status_file}",
        }

    try:
        return json.loads(r["stdout"])
    except Exception:
        return {
            "source_tier": source_tier,
            "status": "invalid_json",
            "raw": r["stdout"][:2000],
        }


def create_restore_control(
    *,
    source_tier: str,
    dataset_id: str,
    restore_mode: str,
    restore_scope: str,
    reason: str,
    target_id: str = "tier1-restore-agent",
    restore_id: str | None = None,
    expires_in_minutes: int = 60,
    approved_by: str = "orchestrator-admin",
    body: dict | None = None,
) -> dict:
    source_tier = source_tier.strip().lower()

    if source_tier not in {"tier4", "tier5"}:
        raise ValueError("source_tier must be tier4 or tier5")

    dataset_id = _restore_validate_id("dataset_id", dataset_id)

    allowed_modes = {
        "tier4": {
            "normal_full_restore": "full",
            "emergency_followup_restore": "medium_low_only",
        },
        "tier5": {
            "emergency_high_restore": "high_only",
        },
    }

    if restore_mode not in allowed_modes[source_tier]:
        raise ValueError(f"invalid restore_mode={restore_mode} for source_tier={source_tier}")

    expected_scope = allowed_modes[source_tier][restore_mode]

    if restore_scope != expected_scope:
        raise ValueError(
            f"restore_scope must be {expected_scope} for restore_mode={restore_mode}"
        )

    restore_id = restore_id or _restore_new_id(f"restore-{source_tier}")
    restore_id = _restore_validate_id("restore_id", restore_id)

    target = _restore_get_target(target_id, source_tier, body=body)

    include_high = restore_scope in {"full", "high_only"}
    include_medium = restore_scope in {"full", "medium_low_only"}
    include_low = restore_scope in {"full", "medium_low_only"}
    skip_existing_high = restore_scope == "medium_low_only"
    if source_tier in ("tier4", "tier5"):
        target_identity_check = [
            "ip",
            "mac",
            "ssh_host_fingerprint",
            "vault_restore_control",
            "sftp_drop_only_user",
            "chroot_drop_zone",
        ]
    else:
        target_identity_check = [
            "ip",
            "mac",
            "hostname",
            "ssh_host_fingerprint",
            "vault_restore_control",
        ]
    control = {
        "restore_id": restore_id,
        "dataset_id": dataset_id,
        "source_tier": source_tier,

        "restore_mode": restore_mode,
        "restore_scope": restore_scope,
        "restore_phase": (
            "phase_1_high_first" if restore_scope == "high_only"
            else "phase_2_medium_low" if restore_scope == "medium_low_only"
            else "normal_full"
        ),

        "include_high": include_high,
        "include_medium": include_medium,
        "include_low": include_low,
        "skip_existing_high": skip_existing_high,

        "target_id": target_id,
        "target_agent_host": target["target_agent_host"],
        "target_agent_ip": target["target_agent_ip"],
        "target_agent_user": target["target_agent_user"],
        "target_agent_mac": target["target_agent_mac"],
        "target_agent_ssh_fingerprint": target["target_agent_ssh_fingerprint"],
        "target_incoming_dir": target["target_incoming_dir"],

        "status": "approved",
        "approved_by": approved_by,
        "approved_from": _get_client_ip(),
        "reason": reason or "restore request approved by Orchestrator",
        "created_at": utc_now().isoformat(),
        "expires_at": _restore_expires_at(expires_in_minutes),
        "single_use": True,

        "orchestrator_policy": {
            "tier4_or_tier5_connect_sql_directly": False,
            "restore_must_go_through_tier1_restore_agent": True,
            "target_identity_check": target_identity_check,
            "sql_import_location": "tier1_restore_agent_sql_staging",
        },
    }

    vault_write_path(_restore_control_path(restore_id), control)

    log.info(
        "restore_control_created",
        restore_id=restore_id,
        dataset_id=dataset_id,
        source_tier=source_tier,
        restore_mode=restore_mode,
        restore_scope=restore_scope,
        target_agent_ip=target["target_agent_ip"],
    )

    return control


def run_restore_release_service(source_tier: str, restore_id: str, timeout: int = 1800) -> dict:
    source_tier = source_tier.strip().lower()
    restore_id = _restore_validate_id("restore_id", restore_id)

    if source_tier not in RESTORE_RELEASE_SERVICES:
        return {
            "ok": False,
            "source_tier": source_tier,
            "error": "source_tier must be tier4 or tier5",
        }

    service = RESTORE_RELEASE_SERVICES[source_tier]

    if source_tier not in TIER_NODES:
        return {
            "ok": False,
            "source_tier": source_tier,
            "service": service,
            "error": f"{source_tier} is not configured in TIER_NODES",
        }

    safe_restore_id = shlex.quote(f"RESTORE_ID={restore_id}")
    safe_service = shlex.quote(service)

    cmd = (
        f"sudo -n /usr/bin/systemctl set-environment {safe_restore_id} && "
        f"sudo -n /usr/bin/systemctl start {safe_service}; "
        f"rc=$?; "
        f"sudo -n /usr/bin/systemctl unset-environment RESTORE_ID || true; "
        f"exit $rc"
    )

    r = ssh_cmd(source_tier, cmd, timeout=timeout)
    release_status = _restore_read_release_status(source_tier)

    return {
        "ok": r["ok"],
        "source_tier": source_tier,
        "service": service,
        "restore_id": restore_id,
        "stdout": r["stdout"],
        "stderr": r["stderr"],
        "returncode": r["returncode"],
        "release_status": release_status,
    }
def run_tier1_apply_sql_service(mode: str, restore_id: str = "", timeout: int = 1800) -> dict:
    mode = (mode or "").strip()
    restore_id = (restore_id or "").strip()

    allowed_modes = {
        "latest_tier4_full",
        "latest_tier5_high",
        "latest_tier4_medium_low",
    }

    if mode not in allowed_modes:
        return {
            "ok": False,
            "service": TIER1_APPLY_SQL_SERVICE,
            "returncode": 2,
            "stdout": "",
            "stderr": f"Invalid mode: {mode}",
            "apply_status": {"status": "invalid_mode"},
        }

    safe_mode = shlex.quote(f"MODE={mode}")

    env_cmd = f"sudo -n /usr/bin/systemctl set-environment {safe_mode} && "

    if restore_id:
        restore_id = _restore_validate_id("restore_id", restore_id)
        safe_restore_id = shlex.quote(f"RESTORE_ID={restore_id}")
        env_cmd += f"sudo -n /usr/bin/systemctl set-environment {safe_restore_id} && "

    safe_service = shlex.quote(TIER1_APPLY_SQL_SERVICE)

    cmd = (
        env_cmd
        + f"sudo -n /usr/bin/systemctl start {safe_service}; "
        + "rc=$?; "
        + "sudo -n /usr/bin/systemctl unset-environment MODE || true; "
        + "sudo -n /usr/bin/systemctl unset-environment RESTORE_ID || true; "
        + "exit $rc"
    )

    r = ssh_cmd("tier1", cmd, timeout=timeout)

    status = {}
    sr = ssh_cmd("tier1", f"cat {shlex.quote(TIER1_APPLY_SQL_STATUS_FILE)}", timeout=10)

    if sr.get("ok"):
        try:
            status = json.loads(sr["stdout"])
        except Exception:
            status = {"status": "invalid_json", "raw": sr["stdout"][:2000]}
    else:
        status = {
            "status": "unknown",
            "message": sr.get("stderr") or sr.get("stdout") or "cannot read Tier1 SQL apply status file",
        }

    return {
        "ok": r["ok"],
        "service": TIER1_APPLY_SQL_SERVICE,
        "mode": mode,
        "restore_id": restore_id,
        "stdout": r["stdout"],
        "stderr": r["stderr"],
        "returncode": r["returncode"],
        "apply_status": status,
    }

def _restore_chain_requires_medium_low(dataset_id: str) -> tuple[bool, dict]:
    chain = _restore_chain_get(dataset_id)
    high_done = bool(chain.get("high_restore_done", False))
    medium_low_done = bool(chain.get("medium_low_completed", False))
    active = bool(chain.get("hospital_failure_restore_active", False))

    if high_done and active and not medium_low_done:
        return True, chain

    return False, chain

def _restore_request_common(
    *,
    source_tier: str,
    restore_mode: str,
    restore_scope: str,
    default_run_now: bool,
):
    body = request.get_json(silent=True) or {}

    dataset_id = (body.get("dataset_id") or body.get("dataset") or "").strip()
    if not dataset_id:
        return ok({"error": "dataset_id is required"}, 400)

    restore_id = (body.get("restore_id") or "").strip() or None
    reason = (body.get("reason") or f"{restore_mode} requested from dashboard").strip()
    target_id = (body.get("target_id") or "tier1-restore-agent").strip()
    expires_in_minutes = _restore_int(body.get("expires_in_minutes"), 60)
    run_now = _restore_bool(body.get("run_now"), default_run_now)

    try:
        control = create_restore_control(
            source_tier=source_tier,
            dataset_id=dataset_id,
            restore_mode=restore_mode,
            restore_scope=restore_scope,
            reason=reason,
            target_id=target_id,
            restore_id=restore_id,
            expires_in_minutes=expires_in_minutes,
            approved_by="orchestrator-admin",
            body=body,
        )
    except Exception as exc:
        log.error(
            "restore_control_create_failed",
            error=str(exc),
            tb=traceback.format_exc(),
        )
        return ok({"status": "failed", "error": str(exc)}, 400)

    response = {
        "status": "restore_control_created",
        "restore_id": control["restore_id"],
        "control_path": _restore_control_path(control["restore_id"]),
        "control": control,
    }

    if run_now:
        release = run_restore_release_service(source_tier, control["restore_id"])
        response["release"] = release

        if not release.get("ok"):
            response["status"] = "release_failed"
            return ok(response, 500)

        # Ghi restore-chain để ép đúng thứ tự:
        # Tier5 high xong thì Tier4 chỉ được medium-low.
        if source_tier == "tier5" and restore_scope == "high_only":
            response["restore_chain"] = _restore_chain_mark_tier5_high_done(
                dataset_id=dataset_id,
                restore_id=control["restore_id"],
                release=release,
            )

        if source_tier == "tier4" and restore_scope == "medium_low_only":
            response["restore_chain"] = _restore_chain_mark_medium_low_done(
                dataset_id=dataset_id,
                restore_id=control["restore_id"],
                release=release,
            )

        response["status"] = "release_completed"

    return ok(response)

@app.route("/incident/hospital-failure/restore", methods=["POST"])
@require_auth
def incident_hospital_failure_restore():
    body = request.get_json(silent=True) or {}

    dataset_id = (body.get("dataset_id") or body.get("dataset") or "").strip()
    if not dataset_id:
        return ok({"error": "dataset_id is required"}, 400)

    dataset_id = _restore_validate_id("dataset_id", dataset_id)

    expires_in_minutes = _restore_int(body.get("expires_in_minutes"), 60)
    reason = (body.get("reason") or "hospital failure restore: Tier5 high first, Tier4 medium-low follow-up").strip()

    response = {
        "status": "running",
        "scenario": "hospital_failure",
        "dataset_id": dataset_id,
        "policy": {
            "phase_1": "Tier5 restores HIGH Critical first",
            "phase_2": "Tier4 restores MEDIUM + LOW only",
            "no_blind_high_overwrite": True,
            "sql_import_location": "Tier1 Restore Agent SQL staging",
        },
        "phases": [],
    }

    # Phase 1: Tier5 HIGH
    try:
        phase1_control = create_restore_control(
            source_tier="tier5",
            dataset_id=dataset_id,
            restore_mode="emergency_high_restore",
            restore_scope="high_only",
            reason=f"{reason} | phase 1 high from Tier5",
            target_id=(body.get("target_id") or "tier1-restore-agent").strip(),
            expires_in_minutes=expires_in_minutes,
            approved_by="orchestrator-admin",
            body=body,
        )
    except Exception as exc:
        return ok({
            **response,
            "status": "failed",
            "failed_at": "phase_1_create_control",
            "error": str(exc),
        }, 400)

    phase1_release = run_restore_release_service("tier5", phase1_control["restore_id"])

    response["phases"].append({
        "phase": "phase_1_high_first",
        "source_tier": "tier5",
        "restore_id": phase1_control["restore_id"],
        "control": phase1_control,
        "release": phase1_release,
    })

    if not phase1_release.get("ok"):
        response["status"] = "failed"
        response["failed_at"] = "phase_1_tier5_high"
        return ok(response, 500)

    chain = _restore_chain_mark_tier5_high_done(
        dataset_id=dataset_id,
        restore_id=phase1_control["restore_id"],
        release=phase1_release,
    )

    # Phase 2: Tier4 MEDIUM + LOW only
    try:
        phase2_control = create_restore_control(
            source_tier="tier4",
            dataset_id=dataset_id,
            restore_mode="emergency_followup_restore",
            restore_scope="medium_low_only",
            reason=f"{reason} | phase 2 medium-low from Tier4, skip existing high",
            target_id=(body.get("target_id") or "tier1-restore-agent").strip(),
            expires_in_minutes=expires_in_minutes,
            approved_by="orchestrator-admin",
            body=body,
        )
    except Exception as exc:
        return ok({
            **response,
            "status": "failed",
            "failed_at": "phase_2_create_control",
            "restore_chain": chain,
            "error": str(exc),
        }, 400)

    phase2_release = run_restore_release_service("tier4", phase2_control["restore_id"])

    response["phases"].append({
        "phase": "phase_2_medium_low",
        "source_tier": "tier4",
        "restore_id": phase2_control["restore_id"],
        "control": phase2_control,
        "release": phase2_release,
    })

    if not phase2_release.get("ok"):
        response["status"] = "failed"
        response["failed_at"] = "phase_2_tier4_medium_low"
        response["restore_chain"] = chain

    )

    response["restore_chain"] = chain

    return ok(response)

def _demo_read_json_file(path: Path, default: dict | None = None) -> dict:
    default = default or {}
    try:
        if path.exists():
            return json.loads(path.read_text(encoding="utf-8"))
    except Exception:
        pass
    return dict(default)


def _demo_write_json_file(path: Path, data: dict) -> dict:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(
        json.dumps(data, indent=2, ensure_ascii=False, default=str),
        encoding="utf-8",
    )
    return data

def _demo_int(v: Any, default: int = 0) -> int:
    try:
        if v is None or v == "":
            return default
        return int(v)
    except Exception:
        return default


def _demo_counts_from_tier_status(status: dict) -> dict:
    """
    Lấy baseline từ latest_status.json của Tier1.
    Ưu tiên schema mới:
      classification_summary.total_export_items/high_items/medium_items/low_items
    Fallback schema cũ:
      total_patients/high_patients/medium_patients/low_patients
    """
    status = status or {}
    cs = status.get("classification_summary", {}) or {}

    high = _demo_int(
        cs.get("high_items", cs.get("high_rows", cs.get("high_patients", 0))),
        0,
    )
    medium = _demo_int(
        cs.get("medium_items", cs.get("medium_rows", cs.get("medium_patients", 0))),
        0,
    )
    low = _demo_int(
        cs.get("low_items", cs.get("low_rows", cs.get("low_patients", 0))),
        0,
    )

    total = _demo_int(
        cs.get("total_export_items", cs.get("total_rows", cs.get("total_patients", 0))),
        0,
    )

    if total <= 0:
        total = high + medium + low

    return {
        "total": total,
        "high": high,
        "medium": medium,
        "low": low,
    }


def _demo_counts_from_sql_source(sql_source: dict) -> dict:
    """
    /status/sql-source trả về:
      {
        sql_server: {
          export_item_count: ...,
          classification_summary: {
            high_items, medium_items, low_items
          }
        }
      }
    """
    ss = (sql_source or {}).get("sql_server", {}) or {}
    cs = ss.get("classification_summary", {}) or {}

    high = _demo_int(cs.get("high_items", cs.get("high_rows", 0)), 0)
    medium = _demo_int(cs.get("medium_items", cs.get("medium_rows", 0)), 0)
    low = _demo_int(cs.get("low_items", cs.get("low_rows", 0)), 0)

    total = _demo_int(ss.get("export_item_count", ss.get("current_total", 0)), 0)
    if total <= 0:
        total = high + medium + low

    return {
        "total": total,
        "high": high,
        "medium": medium,
        "low": low,
    }


def _demo_get_sql_source_status() -> dict:
    """
    Gọi trực tiếp script Tier1 giống endpoint /status/sql-source.
    Không gọi HTTP nội bộ để tránh loop Flask.
    """
    r = ssh_cmd(
        "tier1",
        "/opt/hospital-app/scripts/sql_source_status.sh",
        timeout=90,
    )

    if not r.get("ok"):
        return {
            "ok": False,
            "status": "failed",
            "error": r.get("stderr") or r.get("stdout") or "cannot read sql source",
            "raw": r,
        }

    try:
        data = json.loads(r["stdout"])
        data["ok"] = True
        return data
    except Exception:
        return {
            "ok": False,
            "status": "invalid_json",
            "raw": r.get("stdout", ""),
            "stderr": r.get("stderr", ""),
        }

def _demo_build_sql_health(auto_capture_baseline: bool = True) -> dict:
    baseline = _demo_get_baseline(auto_capture=auto_capture_baseline)
    sql_source = _demo_get_sql_source_status()

    if not sql_source.get("ok"):
        return {
            "status": "sql_status_unavailable",
            "severity": "unknown",
            "message": "Cannot read SQL source status from Tier1.",
            "baseline": baseline,
            "sql_source_error": sql_source,
            "updated_at": _demo_now(),
        }

    expected = baseline.get("expected", {}) or {}
    current = _demo_counts_from_sql_source(sql_source)

    exp_total = _demo_int(expected.get("total"), 0)
    exp_high = _demo_int(expected.get("high"), 0)
    exp_medium = _demo_int(expected.get("medium"), 0)
    exp_low = _demo_int(expected.get("low"), 0)

    cur_total = _demo_int(current.get("total"), 0)
    cur_high = _demo_int(current.get("high"), 0)
    cur_medium = _demo_int(current.get("medium"), 0)
    cur_low = _demo_int(current.get("low"), 0)

    missing = {
        "total": max(exp_total - cur_total, 0),
        "high": max(exp_high - cur_high, 0),
        "medium": max(exp_medium - cur_medium, 0),
        "low": max(exp_low - cur_low, 0),
    }

    # Logic dashboard:
    # - current == expected       => healthy
    # - 0 < current < expected    => warning
    # - current == 0              => critical
    # - current mất gần hết >80%  => critical
    if exp_total <= 0:
        status = "baseline_missing"
        severity = "unknown"
        recommended_action = "capture_baseline_after_full_pipeline"
        message = "No valid baseline. Run full pipeline and capture baseline first."
    elif cur_total == exp_total and missing["total"] == 0:
        status = "healthy"
        severity = "normal"
        recommended_action = "none"
        message = "SQL data matches the latest backup baseline."
    elif cur_total == 0:
        status = "critical_full_loss"
        severity = "critical"
        recommended_action = "restore_emergency_tier5_high_first"
        message = "SQL source appears empty. Use emergency restore: Tier5 High first, then Tier4 Medium/Low."
    elif cur_total < exp_total and missing["total"] >= int(exp_total * 0.8):
        status = "critical_near_full_loss"
        severity = "critical"
        recommended_action = "restore_emergency_tier5_high_first"
        message = "Most SQL data is missing. Use emergency restore."
    elif cur_total < exp_total:
        status = "warning_partial_loss"
        severity = "warning"
        recommended_action = "restore_normal_tier4"
        message = "Partial SQL data loss detected. Restore missing rows from Tier4."
    elif cur_total > exp_total:
        status = "warning_unexpected_extra_rows"
        severity = "warning"
        recommended_action = "review_sql_source"
        message = "SQL has more rows than the baseline. Review whether new live data was inserted after backup."
    else:
        status = "unknown"
        severity = "unknown"
        recommended_action = "review"
        message = "SQL health could not be classified."

    return {
        "status": status,
        "severity": severity,
        "message": message,
        "dataset_id": baseline.get("baseline_dataset_id", ""),
        "expected": {
            "total": exp_total,
            "high": exp_high,
            "medium": exp_medium,
            "low": exp_low,
        },
        "current": {
            "total": cur_total,
            "high": cur_high,
            "medium": cur_medium,
            "low": cur_low,
        },
        "missing": missing,
        "recommended_action": recommended_action,
        "baseline": baseline,
        "sql_source": sql_source.get("sql_server", {}),
        "restore_state": _demo_read_json_file(DEMO_RESTORE_STATE_FILE, {}),
        "updated_at": _demo_now(),
    }

def _demo_latest_restore_dataset_id_from_statuses() -> str:
    """
    Dataset dùng cho restore phải lấy từ backup pipeline mới nhất,
    không lấy từ SQL baseline cũ.

    Thứ tự ưu tiên:
    - Tier5 latest_status nếu accepted/locked.
    - Tier4 latest_status nếu full backup/promote đã xong.
    - Tier2/Tier1 latest_status nếu có dataset mới.
    """
    candidates = []

    status_files = [
        "/data/tier5/logs/latest_status.json",
        "/data/backupstore/logs/latest_status.json",
        "/data/staging/app/logs/latest_status.json",
        "/var/log/backup_export/latest_status.json",
    ]

    # Các file này có thể chỉ tồn tại trên remote tier, nên nếu orchestrator
    # đã có hàm đọc status qua SSH thì phần route/status vẫn xử lý.
    # Ở đây chỉ dùng helper local-safe, không fail nếu file không tồn tại.
    for fp in status_files:
        try:
            data = _demo_read_json_file(fp, {})
            ds = (data.get("dataset_id") or data.get("active_dataset_id") or "").strip()
            st = (data.get("status") or "").strip().lower()
            step = (data.get("current_step") or "").strip().lower()
            if ds and st in {"success", "completed", "accepted", "connected"}:
                candidates.append(ds)
            elif ds and any(x in step for x in [
                "accepted",
                "locked",
                "promote_completed",
                "full_backup_stored",
                "transfer_to_tier2_completed",
                "snapshot_and_sync_to_tier4_completed"
            ]):
                candidates.append(ds)
        except Exception:
            pass

    # Fallback mạnh hơn: lấy từ demo payload visible statuses nếu hàm có sẵn.
    try:
        statuses = _demo_collect_statuses()
        for key in ["tier5", "tier4", "tier2", "tier1"]:
            item = statuses.get(key, {}) or {}
            ds = (item.get("dataset_id") or "").strip()
            st = (item.get("status") or "").strip().lower()
            step = (item.get("current_step") or "").strip().lower()
            if ds and (st == "success" or "completed" in step or "accepted" in step or "locked" in step):
                candidates.append(ds)
    except Exception:
        pass

    # bỏ trùng, lấy ứng viên cuối cùng/mới nhất theo thứ tự ưu tiên thực tế
    seen = []
    for ds in candidates:
        if ds and ds not in seen:
            seen.append(ds)

    return seen[0] if seen else ""


def _demo_get_restore_dataset_id(body: dict | None = None) -> str:
    """
    Dataset cho restore.
    Không dùng SQL baseline làm lựa chọn đầu tiên, vì baseline có thể là lần cũ.
    """
    body = body or {}

    body_ds = (
        body.get("dataset_id")
        or body.get("restore_dataset_id")
        or body.get("active_dataset_id")
        or ""
    )
    body_ds = str(body_ds).strip()
    if body_ds:
        return body_ds

    latest_ds = _demo_latest_restore_dataset_id_from_statuses()
    if latest_ds:
        return latest_ds

    # Fallback cuối cùng: dùng hàm cũ nếu có.
    try:
        return _demo_get_dataset_id(body)
    except Exception:
        return ""

def _demo_get_dataset_id(body: dict | None = None) -> str:
    body = body or {}

    dataset_id = (
        body.get("dataset_id")
        or body.get("dataset")
        or request.args.get("dataset_id", "")
        or ""
    )
    dataset_id = str(dataset_id).strip()

    if dataset_id:
        return _restore_validate_id("dataset_id", dataset_id)

    baseline = _demo_get_baseline(auto_capture=True)
    dataset_id = str(baseline.get("baseline_dataset_id", "")).strip()

    if dataset_id:
        return _restore_validate_id("dataset_id", dataset_id)

    # Fallback cuối cùng: lấy từ tier1/tier4 status
    for tier in ("tier4", "tier1"):
        st = read_tier_status(tier)
        dataset_id = str((st or {}).get("dataset_id", "")).strip()
        if dataset_id:
            return _restore_validate_id("dataset_id", dataset_id)

    raise ValueError("Cannot determine dataset_id. Run full pipeline and capture baseline first.")


def _demo_normal_restore_worker(*, dataset_id: str, restore_id: str) -> None:
    try:
        _demo_update_restore_state(
            status="running",
            mode="normal_tier4_restore",
            phase="release_from_tier4",
            dataset_id=dataset_id,
            restore_id=restore_id,
            message="Releasing full restore package from Tier4 to Tier1.",
            started_at=_demo_now(),
        )

        release = run_restore_release_service("tier4", restore_id)

        _demo_update_restore_state(
            status="running" if release.get("ok") else "failed",
            mode="normal_tier4_restore",
            phase="release_from_tier4_done" if release.get("ok") else "release_from_tier4_failed",
            dataset_id=dataset_id,
            restore_id=restore_id,
            release=release,
            message="Tier4 full restore package released to Tier1." if release.get("ok") else "Tier4 release failed.",
        )

        if not release.get("ok"):
            return

        apply = run_tier1_apply_sql_service(
            mode="latest_tier4_full",
            restore_id=restore_id,
        )

        _invalidate_sql_health_cache()
        health = _demo_build_sql_health_cached(
            force=True,
            auto_capture_baseline=True,
        )

        _demo_update_restore_state(
            status="completed" if apply.get("ok") else "failed",
            mode="normal_tier4_restore",
            phase="sql_apply_done" if apply.get("ok") else "sql_apply_failed",
            dataset_id=dataset_id,
            restore_id=restore_id,
            release=release,
            apply=apply,
            sql_health_after_restore=health,
            message="SQL restore from Tier4 completed." if apply.get("ok") else "Tier1 SQL apply failed.",
            finished_at=_demo_now(),
        )

    except Exception as exc:
        _demo_update_restore_state(
            status="failed",
            mode="normal_tier4_restore",
            phase="exception",
            dataset_id=dataset_id,
            restore_id=restore_id,
            message=str(exc),
            traceback=traceback.format_exc(),
            finished_at=_demo_now(),
        )

    restore_id_tier5: str,

SQL_HEALTH_CACHE = {

SQL_HEALTH_CACHE_LOCK = _threading.Lock()

def _invalidate_sql_health_cache():
    with SQL_HEALTH_CACHE_LOCK:
        SQL_HEALTH_CACHE["ts"] = 0
        SQL_HEALTH_CACHE["data"] = None
        SQL_HEALTH_CACHE["last_error"] = ""
        SQL_HEALTH_CACHE["last_error_at"] = ""


def _demo_sql_health_is_transient_error(health: dict) -> bool:
    """
    Lỗi tạm thời thì không được ghi đè cache tốt.
    Ví dụ: SSH Tier1 lỗi thoáng qua, sql_source_status.sh timeout,
    invalid_json, sql_status_unavailable.
    """
    health = health or {}

    status = str(health.get("status", "")).lower()
    severity = str(health.get("severity", "")).lower()

    if status in {
        "sql_status_unavailable",
        "failed",
        "invalid_json",
        "timeout",
        "ssh_failed",
    }:
        return True

    if severity == "unknown" and not health.get("current"):
        return True

    return False


def _demo_mark_cached_health_warning(health: dict, error: str = "") -> dict:
    """
    Trả lại health cũ nhưng đánh dấu là dùng cache do lần check mới bị lỗi.
    Không làm dashboard nhảy về 0/0.
    """
    data = _json.loads(_json.dumps(health or {}))
    data["cache_status"] = "using_previous_good_sql_health"
    data["last_check_error"] = error or "Latest SQL health check failed temporarily."
    data["last_check_error_at"] = _demo_now()
    return data


def _demo_sql_health_refresh_background(auto_capture_baseline: bool = True) -> None:
    try:
        health = _demo_build_sql_health(
            auto_capture_baseline=auto_capture_baseline
        )

        with SQL_HEALTH_CACHE_LOCK:
            old_good = SQL_HEALTH_CACHE.get("data")

            if _demo_sql_health_is_transient_error(health) and old_good is not None:
                SQL_HEALTH_CACHE["last_error"] = str(
                    health.get("message")
                    or health.get("error")
                    or "Transient SQL health check error"
                )
                SQL_HEALTH_CACHE["last_error_at"] = _demo_now()
                return

            SQL_HEALTH_CACHE["ts"] = _time.time()
            SQL_HEALTH_CACHE["data"] = _json.loads(_json.dumps(health))
            SQL_HEALTH_CACHE["last_error"] = ""
            SQL_HEALTH_CACHE["last_error_at"] = ""

    except Exception as exc:
        with SQL_HEALTH_CACHE_LOCK:
            SQL_HEALTH_CACHE["last_error"] = str(exc)
            SQL_HEALTH_CACHE["last_error_at"] = _demo_now()

        log.warning(
            "sql_health_background_refresh_failed",
            error=str(exc),
        )

    finally:
        with SQL_HEALTH_CACHE_LOCK:
            SQL_HEALTH_REFRESHING["running"] = False


def _demo_build_sql_health_cached(
    *,
    force: bool = False,
    auto_capture_baseline: bool = True,
) -> dict:
    """
    Dashboard dùng hàm này để tránh giật:
    - Cache còn mới: trả cache.
    - Cache hết hạn nhưng có dữ liệu tốt: trả dữ liệu tốt trước, refresh nền.
    - Nếu lần check mới lỗi tạm: không overwrite cache tốt.
    - force=True: check thật, nhưng nếu lỗi tạm vẫn trả cache tốt nếu có.
    """
    now = _time.time()

    with SQL_HEALTH_CACHE_LOCK:
        data = SQL_HEALTH_CACHE.get("data")
        ts = SQL_HEALTH_CACHE.get("ts", 0)
        age = now - ts if ts else 999999

        if (
            not force
            and data is not None
            and age < SQL_HEALTH_CACHE_TTL_SECONDS
        ):
            return _json.loads(_json.dumps(data))

        if not force and data is not None:
            if not SQL_HEALTH_REFRESHING["running"]:
                SQL_HEALTH_REFRESHING["running"] = True
                _threading.Thread(
                    target=_demo_sql_health_refresh_background,
                    kwargs={"auto_capture_baseline": auto_capture_baseline},
                    daemon=True,
                ).start()

            stale = _json.loads(_json.dumps(data))
            stale["cache_status"] = "stale_background_refreshing"
            stale["cache_age_seconds"] = int(age)
            return stale

    try:
        health = _demo_build_sql_health(
            auto_capture_baseline=auto_capture_baseline
        )
    except Exception as exc:
        with SQL_HEALTH_CACHE_LOCK:
            old_good = SQL_HEALTH_CACHE.get("data")

        if old_good is not None:
            return _demo_mark_cached_health_warning(old_good, str(exc))

        raise

    with SQL_HEALTH_CACHE_LOCK:
        old_good = SQL_HEALTH_CACHE.get("data")

        if _demo_sql_health_is_transient_error(health) and old_good is not None:
            SQL_HEALTH_CACHE["last_error"] = str(
                health.get("message")
                or health.get("error")
                or "Transient SQL health check error"
            )
            SQL_HEALTH_CACHE["last_error_at"] = _demo_now()

            return _demo_mark_cached_health_warning(
                old_good,
                SQL_HEALTH_CACHE["last_error"],
            )

        SQL_HEALTH_CACHE["ts"] = _time.time()
        SQL_HEALTH_CACHE["data"] = _json.loads(_json.dumps(health))
        SQL_HEALTH_CACHE["last_error"] = ""
        SQL_HEALTH_CACHE["last_error_at"] = ""

    return health

def _validate_manual_tier_run(tier, state):
    """
    Chặn chạy nhảy tầng, nhưng cho phép emergency route:

    Normal:
      Tier1 → Tier2 → Tier4 → Tier5

    Emergency khi Tier4 sụp:
      Tier1 → Tier2 → Tier5
      Điều kiện: Tier5 control đang emergency, accept_from_tier2=true.
    """
    if tier == "tier1":
        return True, "", ""

    visible = _get_visible_statuses(state)

    # Tier2 cần Tier1 success.
    if tier == "tier2":
        t1 = visible.get("tier1", {}) or {}
        if _status_class(t1.get("status")) != "success":
            return False, "", "Không thể chạy tier2: tier1 chưa SUCCESS trong lần chạy hiện tại."

        ds1 = t1.get("dataset_id", "")
        if not ds1:
            return False, "", "Không thể chạy tier2: tier1 chưa có dataset_id hợp lệ."

        return True, ds1, ""

    # Tier4 cần Tier1 + Tier2 success cùng dataset.
    if tier == "tier4":
        if _tier5_accepts_tier2_emergency():
            return (
                False,
                "",
                "Tier4 đang bị khóa vì Orchestrator đã kích hoạt emergency route Tier2 → Tier5. Hãy reset normal sau khi sự cố được xử lý."
            )
        t1 = visible.get("tier1", {}) or {}
        t2 = visible.get("tier2", {}) or {}

        for req, data in [("tier1", t1), ("tier2", t2)]:
            if _status_class(data.get("status")) != "success":
                return False, "", f"Không thể chạy tier4: {req} chưa SUCCESS trong lần chạy hiện tại."

        ds1 = t1.get("dataset_id", "")
        ds2 = t2.get("dataset_id", "")

        if not ds1 or not ds2:
            return False, "", "Không thể chạy tier4: tier1/tier2 chưa có dataset_id hợp lệ."

        if ds1 != ds2:
            return False, "", f"Không thể chạy tier4: tier1 và tier2 không cùng dataset: {ds1} != {ds2}"

        return True, ds2, ""

    # Tier5 có 2 đường:
    # 1. Normal: Tier1 + Tier2 + Tier4 success cùng dataset.
    # 2. Emergency: Tier1 + Tier2 success cùng dataset, Tier5 đang accept_from_tier2.
    if tier == "tier5":
        t1 = visible.get("tier1", {}) or {}
        t2 = visible.get("tier2", {}) or {}
        t4 = visible.get("tier4", {}) or {}

        for req, data in [("tier1", t1), ("tier2", t2)]:
            if _status_class(data.get("status")) != "success":
                return False, "", f"Không thể chạy tier5: {req} chưa SUCCESS trong lần chạy hiện tại."

        ds1 = t1.get("dataset_id", "")
        ds2 = t2.get("dataset_id", "")

        if not ds1 or not ds2:
            return False, "", "Không thể chạy tier5: tier1/tier2 chưa có dataset_id hợp lệ."

        if ds1 != ds2:
            return False, "", f"Không thể chạy tier5: tier1 và tier2 không cùng dataset: {ds1} != {ds2}"

        # Normal route: Tier4 success thì dùng dataset của Tier4.
        if _status_class(t4.get("status")) == "success":
            ds4 = t4.get("dataset_id", "")
            if not ds4:
                return False, "", "Không thể chạy tier5: tier4 chưa có dataset_id hợp lệ."

            if ds4 != ds2:
                return False, "", f"Không thể chạy tier5: tier4 không cùng dataset với tier2: {ds4} != {ds2}"

            return True, ds4, ""

        # Emergency route: Tier4 không success nhưng Orchestrator đã bật Tier2 → Tier5.
        if _tier5_accepts_tier2_emergency():
            return True, ds2, ""

        return (
            False,
            "",
            "Không thể chạy tier5: tier4 chưa SUCCESS và Tier5 chưa bật emergency accept_from_tier2."
        )

    return False, "", f"Invalid tier: {tier}"

    return {
        "tier": tier,

def _pipeline_steps_before_tier(state, tier):
    """
    Lấy lại các step SUCCESS trước tầng đang chạy.
    Ví dụ chạy Tier2 thì giữ Tier1.
    Chạy Tier4 thì giữ Tier1 + Tier2.
    Chạy Tier5 thì giữ Tier1 + Tier2 + Tier4.
    """
    result = []

    last_run = state.get("last_run", {}) or {}
    old_steps = last_run.get("steps", []) or {}

    old_steps_by_tier = {}
    if isinstance(old_steps, list):
        for step in old_steps:
            if not isinstance(step, dict):
                continue

            tid = step.get("tier", "")
            if tid in PIPELINE_TIERS:
                old_steps_by_tier[tid] = dict(step)

    visible = _get_visible_statuses(state)

    for tid in _required_tiers_before(tier):
        old_step = old_steps_by_tier.get(tid)

        if old_step and _status_class(old_step.get("status")) == "success":
            result.append(old_step)
            continue

        status_data = visible.get(tid, {}) or {}
        status_cls = _status_class(status_data.get("status"))

        if status_cls == "success":
            result.append(
                _status_to_pipeline_step(
                    tid,
                    status_data,
                    fallback_detail=f"{TIER_CONFIG[tid]['title']} completed successfully"
                )
            )

        elif tier == "tier5" and tid == "tier4" and status_cls == "bypassed":
            result.append({
                "tier": "tier4",
                "service": TIER_CONFIG["tier4"]["service"],
                "started_at": status_data.get("updated_at", ""),
                "finished_at": status_data.get("updated_at", ""),
                "status": "bypassed",
                "dataset_id": status_data.get("dataset_id", ""),
                "detail": status_data.get(
                    "message",
                    "Tier4 bypassed because emergency route Tier2 → Tier5 was activated."
                ),
            })

    return result


def _pipeline_dataset_from_steps(steps, fallback=""):
    if fallback:
        return fallback

    for step in steps:
        ds = step.get("dataset_id", "")
        if ds:
            return ds

    return ""


def _build_manual_chain_pipeline(
    state,
    tier,
    reason,
    pipeline_status,
    current_tier,
    dataset_id,
    current_step,
):
    """
    Tạo last_run dạng chuỗi.
    Không ghi đè mất tầng trước.
    Chỉ xóa tầng hiện tại và tầng sau khi chạy lại.
    """
    previous_steps = _pipeline_steps_before_tier(state, tier)
    steps = previous_steps + [current_step]

    old_last_run = state.get("last_run", {}) or {}

    if previous_steps:
        started_at = old_last_run.get("started_at", "") or previous_steps[0].get("started_at", "") or _demo_now()
    else:
        started_at = current_step.get("started_at", "") or _demo_now()

    pipeline = {
        "mode": "manual_chain",
        "tier": tier,
        "reason": reason,
        "status": pipeline_status,
        "current_tier": current_tier,
        "started_at": started_at,
        "finished_at": "" if pipeline_status == "running" else _demo_now(),
        "dataset_id": _pipeline_dataset_from_steps(steps, dataset_id),
        "steps": steps,
    }

    if pipeline_status == "failed":
        pipeline["failed_at"] = tier

    return pipeline

def _ssh_command(tier, command, timeout=900):
    cfg = TIER_CONFIG[tier]
    target = f"{cfg['user']}@{cfg['host']}"
    cmd = [
        "ssh",
        "-i", SSH_KEY,
        "-o", "BatchMode=yes",
        "-o", "StrictHostKeyChecking=no",
        "-o", "ConnectTimeout=3",
        target,
        command,
    ]
    p = _subprocess.run(
        cmd,
        text=True,
        capture_output=True,
        timeout=timeout,
    )
    return {
        "ok": p.returncode == 0,
        "returncode": p.returncode,
        "stdout": p.stdout.strip(),
        "stderr": p.stderr.strip(),
        "command": command,
    }


def _read_tier_status(tier):
    cfg = TIER_CONFIG[tier]
    status_file = cfg["status_file"]
    r = _ssh_command(
        tier,
        f"cat {_shlex.quote(status_file)}",
        timeout=8,
    )
    if not r["ok"]:
        return {
            "tier": tier,
            "tier_name": cfg["title"],
            "status": "offline",
            "current_step": "cannot_read_status",
            "dataset_id": "",
            "message": r["stderr"] or f"Cannot read {status_file}",
            "updated_at": "",
        }

    try:
        data = _json.loads(r["stdout"])
        if not isinstance(data, dict):
            raise ValueError("status json is not object")
        data.setdefault("tier", tier)
        data.setdefault("tier_name", cfg["title"])
        data.setdefault("status", "unknown")
        data.setdefault("current_step", "")
        data.setdefault("dataset_id", "")
        data.setdefault("message", "")
        data.setdefault("updated_at", "")
        return data
    except Exception as exc:
        return {
            "tier": tier,
            "tier_name": cfg["title"],
            "status": "invalid_json",
            "current_step": "invalid_json",
            "dataset_id": "",
            "message": f"Invalid JSON in {status_file}: {exc}",
            "updated_at": "",
            "raw": r["stdout"][:1000],
        }


def _short(v, n=22):
    if v is None:
        return ""
    v = str(v)
    if len(v) <= n:
        return v
    return v[:n] + "..."


def _status_class(status):
    s = str(status or "unknown").lower()

    if s in ["success", "connected", "ok", "accepted_and_locked"]:
        return "success"

    if s in ["running", "processing"]:
        return "running"

    if s in ["failed", "error", "invalid_json"]:
        return "failed"

    if s in ["not_run", "waiting_for_new_run"]:
        return "not_run"

    if s in ["bypassed", "skipped", "emergency_bypass"]:
        return "bypassed"

    if s in ["offline", "unknown"]:
        return "unknown"

    return s

def _collect_all_statuses(force_refresh=False):
    now = _time.time()

    if (
        not force_refresh
        and STATUS_CACHE["data"] is not None
        and now - STATUS_CACHE["ts"] < STATUS_CACHE_TTL_SECONDS
    ):
        return _json.loads(_json.dumps(STATUS_CACHE["data"]))

    tiers = ["tier1", "tier2", "tier4", "tier5"]
    statuses = {}

    with _ThreadPoolExecutor(max_workers=4) as executor:
        future_map = {
            executor.submit(_read_tier_status, tid): tid
            for tid in tiers
        }

        for future, tid in future_map.items():
            try:
                statuses[tid] = future.result(timeout=8)
            except Exception as exc:
                statuses[tid] = {
                    "tier": tid,
                    "tier_name": TIER_CONFIG[tid]["title"],
                    "status": "offline",
                    "current_step": "read_status_timeout",
                    "dataset_id": "",
                    "message": f"Cannot read tier status: {exc}",
                    "updated_at": "",
                }

    STATUS_CACHE["ts"] = now
    STATUS_CACHE["data"] = _json.loads(_json.dumps(statuses))

    return statuses

def _demo_restore_parse_ts(value):
    from datetime import datetime, timezone

    if not value or not isinstance(value, str):
        return None

    v = value.strip()

    try:
        if v.endswith("Z"):
            v = v[:-1] + "+00:00"
        return datetime.fromisoformat(v)
    except Exception:
        pass

    try:
        return datetime.strptime(v[:19], "%Y-%m-%dT%H:%M:%S").replace(tzinfo=timezone.utc)
    except Exception:
        return None


def _demo_restore_duration_label(seconds):
    try:
        seconds = int(round(float(seconds)))
    except Exception:
        seconds = 0

    if seconds < 0:
        seconds = 0

    h = seconds // 3600
    m = (seconds % 3600) // 60
    sec = seconds % 60

    if h > 0:
        return f"{h}h {m}m {sec}s"
    if m > 0:
        return f"{m}m {sec}s"
    return f"{sec}s"


def _demo_restore_metric_int(*values):
    for v in values:
        if v is None:
            continue
        try:
            txt = str(v).strip()
            if txt == "" or txt.lower() in {"none", "null", "nan", "-"}:
                continue
            return int(float(txt))
        except Exception:
            continue
    return 0


def _demo_restore_nested(obj, path, default=None):
    cur = obj
    try:
        for key in path:
            if not isinstance(cur, dict):
                return default
            cur = cur.get(key)
        return cur if cur is not None else default
    except Exception:
        return default


def _demo_enrich_restore_state_for_dashboard(restore_state, current_sql_health=None):
    """
    Enrich restore_state để dashboard hiển thị đúng:
    - thời gian hoàn thành restore
    - số dữ liệu đã khôi phục
    - tốc độ restore

    Fix chính:
    - Nếu total.current bị thiếu/0 nhưng high+medium+low có số liệu thì lấy tổng high+medium+low.
    - Nếu restore completed và missing=0 thì lấy expected_total làm restored_total.
    - Không để speed_label = '-' khi đã có duration và restored_total.
    """
    if not isinstance(restore_state, dict):
        restore_state = {}

    state = dict(restore_state)

    started_at = (
        state.get("started_at")
        or state.get("created_at")
        or state.get("queued_at")
        or ""
    )

    finished_at = (
        state.get("completed_at")
        or state.get("finished_at")
        or state.get("ended_at")
        or ""
    )

    status = str(state.get("status", "not_started") or "not_started").lower()

    if not finished_at and status in {"completed", "success", "done", "finished", "failed", "error"}:
        finished_at = state.get("updated_at", "") or ""

    start_dt = _demo_restore_parse_ts(started_at)
    end_dt = _demo_restore_parse_ts(finished_at)

    if start_dt and not end_dt and status in {"running", "queued", "started", "busy", "in_progress", "processing"}:
        from datetime import datetime, timezone
        end_dt = datetime.now(timezone.utc)

    duration_seconds = 0
    if start_dt and end_dt:
        duration_seconds = max(0, int((end_dt - start_dt).total_seconds()))

    health = (
        state.get("sql_health_after_restore")
        or state.get("final_sql_health")
        or state.get("sql_health_after_final")
        or state.get("sql_health_after_high")
        or current_sql_health
        or {}
    )

    if not isinstance(health, dict):
        health = {}

    total_current = _demo_restore_metric_int(
        _demo_restore_nested(health, ["total", "current"]),
        _demo_restore_nested(health, ["current", "total"]),
        health.get("total_current"),
        health.get("current_total"),
        health.get("restored_total"),
    )

    total_expected = _demo_restore_metric_int(
        _demo_restore_nested(health, ["total", "expected"]),
        _demo_restore_nested(health, ["expected", "total"]),
        health.get("total_expected"),
        health.get("expected_total"),
        health.get("baseline_total"),
    )

    total_missing = _demo_restore_metric_int(
        _demo_restore_nested(health, ["total", "missing"]),
        _demo_restore_nested(health, ["missing", "total"]),
        health.get("total_missing"),
        health.get("missing_total"),
    )

    high_current = _demo_restore_metric_int(
        _demo_restore_nested(health, ["high", "current"]),
        _demo_restore_nested(health, ["current", "high"]),
        _demo_restore_nested(health, ["classification_summary", "high_items"]),
        _demo_restore_nested(health, ["classification_summary", "high_patients"]),
        health.get("high_current"),
        health.get("high_restored"),
    )

    medium_current = _demo_restore_metric_int(
        _demo_restore_nested(health, ["medium", "current"]),
        _demo_restore_nested(health, ["current", "medium"]),
        _demo_restore_nested(health, ["classification_summary", "medium_items"]),
        _demo_restore_nested(health, ["classification_summary", "medium_patients"]),
        health.get("medium_current"),
        health.get("medium_restored"),
    )

    low_current = _demo_restore_metric_int(
        _demo_restore_nested(health, ["low", "current"]),
        _demo_restore_nested(health, ["current", "low"]),
        _demo_restore_nested(health, ["classification_summary", "low_items"]),
        _demo_restore_nested(health, ["classification_summary", "low_patients"]),
        health.get("low_current"),
        health.get("low_restored"),
    )

    grouped_total = high_current + medium_current + low_current

    if total_expected <= 0:
        total_expected = _demo_restore_metric_int(
            _demo_restore_nested(current_sql_health or {}, ["total", "expected"]),
            _demo_restore_nested(current_sql_health or {}, ["expected", "total"]),
            _demo_restore_nested(current_sql_health or {}, ["baseline", "expected", "total"]),
        )

    if total_missing <= 0:
        total_missing = _demo_restore_metric_int(
            _demo_restore_nested(current_sql_health or {}, ["total", "missing"]),
            _demo_restore_nested(current_sql_health or {}, ["missing", "total"]),
        )

    # Fix quan trọng: nếu total_current = 0 nhưng grouped_total có số liệu thì lấy grouped_total.
    if total_current <= 0 and grouped_total > 0:
        total_current = grouped_total

    # Nếu restore đã completed, expected có số liệu và missing=0 thì coi như đã restore đủ expected.
    if total_current <= 0 and total_expected > 0 and total_missing == 0 and status in {"completed", "success", "done", "finished"}:
        total_current = total_expected

    # Nếu expected chưa có nhưng current có thì expected = current để không hiển thị 0 / 0.
    if total_expected <= 0 and total_current > 0:
        total_expected = total_current

    if total_expected > 0:
        restored_label = f"{total_current} / {total_expected}"
    else:
        restored_label = str(total_current)

    if duration_seconds > 0 and total_current > 0:
        speed = total_current / duration_seconds
        if speed >= 1:
            speed_label = f"{speed:.1f} records/s"
        else:
            speed_label = f"{speed * 60:.1f} records/min"
    else:
        speed_label = "-"

    restore_percent = 0
    if total_expected > 0:
        try:
            restore_percent = min(100, round((total_current / total_expected) * 100, 1))
        except Exception:
            restore_percent = 0

    if restore_percent == int(restore_percent):
        restore_percent_label = f"{int(restore_percent)}%"
    else:
        restore_percent_label = f"{restore_percent}%"

    restore_scope = (
        state.get("restore_scope")
        or state.get("scope")
        or state.get("mode")
        or ""
    )

    state["restore_metrics"] = {
        "status": status,
        "mode": state.get("mode", ""),
        "phase": state.get("phase", ""),
        "dataset_id": state.get("dataset_id", ""),
        "restore_scope": restore_scope,
        "started_at": started_at,
        "completed_at": finished_at,
        "duration_seconds": duration_seconds,
        "duration_label": _demo_restore_duration_label(duration_seconds),
        "restored_total": total_current,
        "expected_total": total_expected,
        "missing_total": total_missing,
        "restored_patients_label": restored_label,
        "speed_label": speed_label,
        "restore_percent": restore_percent,
        "restore_percent_label": restore_percent_label,
        "message": state.get("message", ""),
    }

    return state

def _run_tier_service(tier, dataset_id="", run_started_at=""):
    cfg = TIER_CONFIG[tier]
    service = cfg["service"]
    # Safety gate cho Node-RED demo dashboard:
    # Khi emergency route Tier2 -> Tier5 đang active, Tier4 không được chạy.
    if tier == "tier4" and _tier5_accepts_tier2_emergency():
        status_data = {
            "tier": "tier4",
            "tier_name": TIER_CONFIG["tier4"]["title"],
            "status": "failed",
            "current_step": "blocked_by_emergency_control",
            "dataset_id": dataset_id or "",
            "message": "Tier4 blocked: emergency Tier2 → Tier5 rescue is active. Reset normal before running Tier4.",
            "updated_at": _demo_now(),
            "classification_summary": {
                "total_patients": "",
                "high_patients": "",
                "medium_patients": "",
                "low_patients": ""
            }
        }
        return {
            "tier": tier,
            "service": service,
            "ok": False,
            "returncode": 99,
            "stdout": "",
            "stderr": status_data["message"],
            "status": status_data,
        }

    commands = []

    if dataset_id and tier != "tier1":
        safe_dataset = _shlex.quote(f"PIPELINE_DATASET_ID={dataset_id}")
        commands.append(f"sudo -n /usr/bin/systemctl set-environment {safe_dataset}")

    commands.append(f"sudo -n /usr/bin/systemctl start {service}")

    if dataset_id and tier != "tier1":
        commands.append("sudo -n /usr/bin/systemctl unset-environment PIPELINE_DATASET_ID || true")

    full_cmd = " && ".join(commands)
    r = _ssh_command(tier, full_cmd, timeout=1800)

    status_data = _read_tier_status(tier)

    status_ok = _status_class(status_data.get("status")) == "success"

    dataset_ok = True
    if tier != "tier1" and dataset_id:
        dataset_ok = status_data.get("dataset_id", "") == dataset_id

    fresh_ok = True
    if run_started_at:
        updated_at = status_data.get("updated_at", "") or ""
        fresh_ok = updated_at >= run_started_at

    final_ok = status_ok and dataset_ok and fresh_ok

    if not dataset_ok:
        status_data["status"] = "failed"
        status_data["current_step"] = "dataset_mismatch"
        status_data["message"] = (
            f"{tier} chạy sai dataset. Expected={dataset_id}, "
            f"actual={status_data.get('dataset_id', '')}"
        )

    if not fresh_ok:
        status_data["status"] = "failed"
        status_data["current_step"] = "stale_status"
        status_data["message"] = (
            f"{tier} chưa tạo status mới sau khi bấm Run. "
            f"run_started_at={run_started_at}, updated_at={status_data.get('updated_at', '')}"
        )

    return {
        "tier": tier,
        "service": service,
        "ok": final_ok,
        "returncode": r["returncode"],
        "stdout": r["stdout"],
        "stderr": r["stderr"],
        "status": status_data,
    }

def _run_single_worker(tier, reason, dataset_id=""):
    started_at = _demo_now()

    running_step = {
        "tier": tier,
        "service": TIER_CONFIG[tier]["service"],
        "started_at": started_at,
        "finished_at": "",
        "status": "running",
        "dataset_id": dataset_id,
        "detail": "Service is running from Orchestrator",
    }

    try:
        state = _read_demo_state()

        # Nếu chạy lại Tier1 thì xóa Tier2/Tier4/Tier5.
        # Nếu chạy lại Tier2 thì xóa Tier4/Tier5.
        # Nếu chạy lại Tier4 thì xóa Tier5.
        _clear_downstream_statuses(state, tier)

        visible = _set_visible_status(
            state,
            tier,
            _make_running_status(tier, dataset_id)
        )

        pipeline = _build_manual_chain_pipeline(
            state=state,
            tier=tier,
            reason=reason,
            pipeline_status="running",
            current_tier=tier,
            dataset_id=dataset_id,
            current_step=running_step,
        )

        _save_pipeline_progress(
            pipeline,
            selected_tier=tier,
            visible_statuses=visible,
        )

        result = _run_tier_service(
            tier,
            dataset_id=dataset_id,
            run_started_at=started_at,
        )

        _invalidate_status_cache()

        visible_status = _result_to_visible_status(
            tier,
            result,
            expected_dataset_id=dataset_id,
        )

        final_step = {
            "tier": tier,
            "service": TIER_CONFIG[tier]["service"],
            "started_at": started_at,
            "finished_at": _demo_now(),
            "status": "success" if result["ok"] else "failed",
            "dataset_id": visible_status.get("dataset_id", dataset_id),
            "detail": visible_status.get("message", result.get("stderr", "")),
        }

        state = _read_demo_state()
        visible = _set_visible_status(state, tier, visible_status)

        pipeline = _build_manual_chain_pipeline(
            state=state,
            tier=tier,
            reason=reason,
            pipeline_status="success" if result["ok"] else "failed",
            current_tier="",
            dataset_id=visible_status.get("dataset_id", dataset_id),
            current_step=final_step,
        )

        pipeline["finished_at"] = _demo_now()

        if not result["ok"]:
            pipeline["failed_at"] = tier

        _save_pipeline_progress(
            pipeline,
            selected_tier=tier,
            visible_statuses=visible,
        )

    except Exception as exc:
        failed_status = {
            "tier": tier,
            "tier_name": TIER_CONFIG[tier]["title"],
            "status": "failed",
            "current_step": "orchestrator_worker_error",
            "dataset_id": dataset_id,
            "message": str(exc),
            "updated_at": _demo_now(),
            "classification_summary": {
                "total_patients": "",
                "high_patients": "",
                "medium_patients": "",
                "low_patients": ""
            }
        }

        failed_step = {
            "tier": tier,
            "service": TIER_CONFIG[tier]["service"],
            "started_at": started_at,
            "finished_at": _demo_now(),
            "status": "failed",
            "dataset_id": dataset_id,
            "detail": str(exc),
        }

        state = _read_demo_state()
        visible = _set_visible_status(state, tier, failed_status)

        pipeline = _build_manual_chain_pipeline(
            state=state,
            tier=tier,
            reason=reason,
            pipeline_status="failed",
            current_tier="",
            dataset_id=dataset_id,
            current_step=failed_step,
        )

        pipeline["failed_at"] = tier
        pipeline["finished_at"] = _demo_now()
        pipeline["error"] = str(exc)

        _save_pipeline_progress(
            pipeline,
            selected_tier=tier,
            visible_statuses=visible,
        )

    finally:
        _finish_run()

def _run_full_worker(reason):
    active_dataset_id = ""

    pipeline = {
        "mode": "normal",
        "reason": reason,
        "status": "running",
        "current_tier": "tier1",
        "started_at": _demo_now(),
        "finished_at": "",
        "dataset_id": "",
        "steps": [],
    }

    try:
        state = _read_demo_state()
        state["selected_tier"] = "tier1"
        state["hide_old_status"] = False
        state["reset_at"] = ""
        state["visible_statuses"] = _empty_visible_statuses("Waiting for full pipeline run")
        state["last_run"] = pipeline
        _write_demo_state(state)

        for tier in PIPELINE_TIERS:
            pipeline["current_tier"] = tier

            step = {
                "tier": tier,
                "service": TIER_CONFIG[tier]["service"],
                "started_at": _demo_now(),
                "finished_at": "",
                "status": "running",
                "dataset_id": active_dataset_id,
                "detail": f"{TIER_CONFIG[tier]['title']} is running",
            }

            pipeline["steps"].append(step)

            state = _read_demo_state()
            visible = _set_visible_status(
                state,
                tier,
                _make_running_status(tier, active_dataset_id)
            )
            _save_pipeline_progress(
                pipeline,
                selected_tier=tier,
                visible_statuses=visible,
            )

            result = _run_tier_service(
                tier,
                dataset_id=active_dataset_id,
                run_started_at=step["started_at"],
            )

            _invalidate_status_cache()

            visible_status = _result_to_visible_status(
                tier,
                result,
                expected_dataset_id=active_dataset_id,
            )

            step["finished_at"] = _demo_now()
            step["status"] = "success" if result["ok"] else "failed"
            step["dataset_id"] = visible_status.get("dataset_id", active_dataset_id)
            step["detail"] = visible_status.get("message", result.get("stderr", ""))

            if tier == "tier1" and result["ok"]:
                active_dataset_id = visible_status.get("dataset_id", "")
                pipeline["dataset_id"] = active_dataset_id

            state = _read_demo_state()
            visible = _set_visible_status(state, tier, visible_status)

            _save_pipeline_progress(
                pipeline,
                selected_tier=tier,
                visible_statuses=visible,
            )

            if not result["ok"]:
                pipeline["status"] = "failed"
                pipeline["failed_at"] = tier
                pipeline["current_tier"] = ""
                pipeline["finished_at"] = _demo_now()

                _save_pipeline_progress(
                    pipeline,
                    selected_tier=tier,
                    visible_statuses=visible,
                )
                return

        pipeline["status"] = "success"
        pipeline["current_tier"] = ""
        pipeline["finished_at"] = _demo_now()

        _save_pipeline_progress(pipeline, selected_tier="tier5")

    except Exception as exc:
        pipeline["status"] = "failed"
        pipeline["failed_at"] = pipeline.get("current_tier", "unknown")
        pipeline["current_tier"] = ""
        pipeline["finished_at"] = _demo_now()
        pipeline["error"] = str(exc)

        _save_pipeline_progress(
            pipeline,
            selected_tier=pipeline.get("failed_at", "tier1"),
        )

    finally:
        _finish_run()

# DEMO RESTORE POLICY v2 SAFE OVERRIDE
# ============================================================
# New restore behavior:
# 1) Emergency Restore button only restores High from Tier5.
# 2) Tier4 Restore button decides automatically:
#    - If Tier5 High was restored first -> Tier4 restores Medium/Low only.
#    - If Tier5 High was not restored first -> Tier4 restores FULL.
# 3) No wait_seconds delay.
# ============================================================

def _demo_restore_read_state_safe() -> dict:
    try:
        return _demo_read_json_file(
            DEMO_RESTORE_STATE_FILE,
            {
                "status": "not_started",
                "mode": "",
                "phase": "",
                "dataset_id": "",
                "message": "No restore running.",
                "updated_at": _demo_now(),
            },
        )
    except Exception:
        return {
            "status": "not_started",
            "mode": "",
            "phase": "",
            "dataset_id": "",
            "message": "No restore running.",
            "updated_at": _demo_now(),
        }


def _demo_restore_write_state_safe(data: dict) -> dict:
    import os
    import json
    from pathlib import Path

    data = dict(data or {})
    data["updated_at"] = _demo_now()

    Path(DEMO_RESTORE_STATE_FILE).parent.mkdir(parents=True, exist_ok=True)

    tmp = DEMO_RESTORE_STATE_FILE + ".tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, ensure_ascii=False, default=str)

    os.replace(tmp, DEMO_RESTORE_STATE_FILE)
    return data


def _demo_restore_sql_health_safe() -> dict:
    try:
        return _demo_build_sql_health(auto_capture_baseline=True)
    except Exception as exc:
        return {
            "status": "sql_health_unavailable",
            "severity": "unknown",
            "message": f"Cannot read SQL health: {exc}",
            "expected": {},
            "current": {},
            "missing": {},
        }


def _demo_tier4_should_restore_medium_low(dataset_id: str) -> bool:
    """
    True nếu dataset này đã restore High từ Tier5 trước đó.
    Khi đó Tier4 chỉ cần restore Medium/Low còn lại.
    """
    dataset_id = (dataset_id or "").strip()
    state = _demo_restore_read_state_safe()

    if state.get("dataset_id") != dataset_id:
        return False

    mode = (state.get("mode") or "").strip()
    phase = (state.get("phase") or "").strip()
    status = (state.get("status") or "").strip()

    # Cách 1: dựa vào restore_state.
    if (
        mode in {"emergency_tier5_high_only", "emergency_hospital_failure"}
        and status in {"tier5_high_completed", "running", "completed"}
        and (
            "tier5_high" in phase
            or "high_sql_apply_done" in phase
            or "high_completed" in phase
            or "sql_apply_done" in phase
        )
    ):
        return True

    # Cách 2: fallback dựa vào SQL Health.
    # Nếu SQL hiện có đúng bằng số High thì xem như High đã về,
    # Tier4 chỉ cần trả Medium/Low.
    health = _demo_restore_sql_health_safe()
    expected = health.get("expected", {}) or {}
    current = health.get("current", {}) or {}
    missing = health.get("missing", {}) or {}

    current_total = _demo_int(current.get("total"), 0)
    expected_high = _demo_int(expected.get("high"), 0)
    expected_total = _demo_int(expected.get("total"), 0)
    missing_total = _demo_int(missing.get("total"), 0)

    if (
        expected_high > 0
        and current_total == expected_high
        and expected_total > current_total
        and missing_total > 0
    ):
        return True

    return False


def _run_emergency_hospital_restore_worker(dataset_id: str, wait_seconds: int = 0, reason: str = ""):
    """
    New behavior:
    - Chỉ restore High Critical từ Tier5.
    - Không chờ 60 giây.
    - Không tự chạy Tier4.
    """
    dataset_id = _restore_validate_id("dataset_id", dataset_id)
    reason = (reason or "Dashboard emergency restore: Tier5 High Critical only").strip()

    restore_state = {
        "status": "running",
        "mode": "emergency_tier5_high_only",
        "phase": "tier5_high_release_running",
        "dataset_id": dataset_id,
        "message": "Tier5 High Critical restore started.",
        "started_at": _demo_now(),
        "wait_seconds": 0,
    }
    _demo_restore_write_state_safe(restore_state)

    try:
        control5 = create_restore_control(
            source_tier="tier5",
            dataset_id=dataset_id,
            restore_mode="emergency_high_restore",
            restore_scope="high_only",
            reason=reason,
        )

        rid5 = control5["restore_id"]

        restore_state.update({
            "restore_id_tier5": rid5,
            "control_tier5": control5,
            "phase": "tier5_high_release_running",
            "message": "Tier5 is releasing High Critical package to Tier1.",
        })
        _demo_restore_write_state_safe(restore_state)

        release5 = run_restore_release_service("tier5", rid5)
        restore_state["tier5_high_release"] = release5

        if not release5.get("ok"):
            restore_state.update({
                "status": "failed",
                "phase": "tier5_high_release_failed",
                "message": "Tier5 High release failed.",
            })
            _demo_restore_write_state_safe(restore_state)
            return

        restore_state.update({
            "phase": "tier5_high_sql_apply_running",
            "message": "Tier5 High released. Tier1 is applying High data to SQL.",
        })
        _demo_restore_write_state_safe(restore_state)

        apply5 = run_tier1_apply_sql_service("latest_tier5_high", rid5)
        restore_state["tier5_high_sql_apply"] = apply5

        if not apply5.get("ok"):
            restore_state.update({
                "status": "failed",
                "phase": "tier5_high_sql_apply_failed",
                "message": "Tier1 SQL apply for Tier5 High failed.",
            })
            _demo_restore_write_state_safe(restore_state)
            return

        health = _demo_restore_sql_health_safe()

        restore_state.update({
            "status": "tier5_high_completed",
            "phase": "tier5_high_sql_apply_done",
            "message": "Tier5 High Critical restored. Press Tier4 restore to recover remaining data.",
            "sql_health_after_high": health,
            "completed_at": _demo_now(),
        })
        _demo_restore_write_state_safe(restore_state)

    except Exception as exc:
        restore_state.update({
            "status": "failed",
            "phase": "tier5_high_exception",
            "message": f"Tier5 High restore failed: {exc}",
            "error": str(exc),
        })
        _demo_restore_write_state_safe(restore_state)


def _run_normal_tier4_restore_worker(dataset_id: str, reason: str = ""):
    """
    New behavior:
    - Nếu Tier5 High đã restore trước đó -> Tier4 restore Medium/Low.
    - Nếu chưa restore High từ Tier5 -> Tier4 restore FULL.
    """
    dataset_id = _restore_validate_id("dataset_id", dataset_id)
    reason = (reason or "Dashboard Tier4 restore").strip()

    use_medium_low = _demo_tier4_should_restore_medium_low(dataset_id)

    if use_medium_low:
        restore_mode = "emergency_followup_restore"
        restore_scope = "medium_low_only"
        apply_mode = "latest_tier4_medium_low"
        ui_mode = "tier4_remaining_medium_low_restore"
        ui_message = "Tier4 restoring remaining Medium/Low data after Tier5 High."
    else:
        restore_mode = "normal_full_restore"
        restore_scope = "full"
        apply_mode = "latest_tier4_full"
        ui_mode = "normal_tier4_full_restore"
        ui_message = "Tier4 restoring full backup to SQL."

    restore_state = {
        "status": "running",
        "mode": ui_mode,
        "phase": "tier4_release_running",
        "dataset_id": dataset_id,
        "restore_scope": restore_scope,
        "apply_mode": apply_mode,
        "message": ui_message,
        "started_at": _demo_now(),
    }
    _demo_restore_write_state_safe(restore_state)

    try:
        control4 = create_restore_control(
            source_tier="tier4",
            dataset_id=dataset_id,
            restore_mode=restore_mode,
            restore_scope=restore_scope,
            reason=reason,
        )

        rid4 = control4["restore_id"]

        restore_state.update({
            "restore_id_tier4": rid4,
            "control_tier4": control4,
            "phase": "tier4_release_running",
            "message": f"Tier4 is releasing restore package scope={restore_scope}.",
        })
        _demo_restore_write_state_safe(restore_state)

        release4 = run_restore_release_service("tier4", rid4)
        restore_state["tier4_release"] = release4

        if not release4.get("ok"):
            restore_state.update({
                "status": "failed",
                "phase": "tier4_release_failed",
                "message": f"Tier4 release failed for scope={restore_scope}.",
            })
            _demo_restore_write_state_safe(restore_state)
            return

        restore_state.update({
            "phase": "tier1_sql_apply_running",
            "message": f"Tier4 package released. Tier1 is applying SQL with mode={apply_mode}.",
        })
        _demo_restore_write_state_safe(restore_state)

        apply4 = run_tier1_apply_sql_service(apply_mode, rid4)
        restore_state["tier4_sql_apply"] = apply4

        if not apply4.get("ok"):
            restore_state.update({
                "status": "failed",
                "phase": "tier1_sql_apply_failed",
                "message": f"Tier1 SQL apply failed for mode={apply_mode}.",
            })
            _demo_restore_write_state_safe(restore_state)
            return

        health = _demo_restore_sql_health_safe()

        restore_state.update({
            "status": "completed",
            "phase": "sql_apply_done",
            "message": "SQL restore from Tier4 completed.",
            "sql_health_after_restore": health,
            "completed_at": _demo_now(),
        })
        _demo_restore_write_state_safe(restore_state)

    except Exception as exc:
        restore_state.update({
            "status": "failed",
            "phase": "tier4_restore_exception",
            "message": f"Tier4 restore failed: {exc}",
            "error": str(exc),
        })
        _demo_restore_write_state_safe(restore_state)
